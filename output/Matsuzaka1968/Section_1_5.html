theorem AC:
  assumes "∀l ∈ Λ. A l ≠ {}"
  shows "(Π l ∈ Λ. A l) ≠ {}"
proof -
  let ?a = "λl. if l ∈ Λ then (SOME al. al ∈ A l) else undefined"
  {
    fix l
    assume "l ∈ Λ"
    with assms have "A l ≠ {}" by auto
    with ‹l ∈ Λ› have "?a l ∈ A l" by (auto intro: some_in_eq[THEN iffD2])
  }
  hence "?a ∈ (Π l ∈ Λ. A l)" by simp
  thus ?thesis by auto
qed

theorem thm_1_7_a_a:
  assumes "f ` A = B"
  obtains s where "s ` B ⊆ A"
    and "id_on (f ∘ s) B"
proof -
  {
    fix b
    assume "b ∈ B"
    with assms(1) obtain a where "a ∈ A" and "b = f a" by auto
    hence "a ∈ f -` {b}" by auto
    with ‹a ∈ A› have "(f -` {b}) ∩ A ≠ {}" by auto
  }
  then obtain s where *: "s ∈ (Π b ∈ B. (f -` {b}) ∩ A)" by (rule AC_E)
  have "s ` B ⊆ A"
  proof (rule subsetI)
    fix a
    assume "a ∈ s ` B"
    then obtain b where "b ∈ B" and "a = s b" by auto
    from this(1) and * have "s b ∈ A" by auto
    with ‹a = s b› show "a ∈ A" by simp
  qed
  moreover have "id_on (f ∘ s) B"
  proof (rule id_onI)
    fix b
    assume "b ∈ B"
    with * have "s b ∈ f -` {b}" by auto
    thus "(f ∘ s) b = b" by simp
  qed
  ultimately show "thesis" by (fact that)
qed

theorem thm_1_7_b_a:
  assumes "A = {} ⟹ B = {}" ― ‹This assumption is not specified in the book. However, there
                                  exists a counterexample without it.›
    ― ‹The assumption @{prop "f ` A ⊆ B"} is not necessary.›
    and "inj_on f A"
  obtains r where "r ` B ⊆ A"
    and "id_on (r ∘ f) A"
proof -
  {
    assume "A = {}"
    with assms(1) have "B = {}" by simp
    let ?r = "λb. undefined"
    from ‹A = {}› and ‹B = {}› have "?r ` B ⊆ A" by simp
    moreover from ‹A = {}› have "id_on (?r ∘ f) A" by simp
    ultimately have "thesis" by (fact that)
  }
  moreover {
    assume "A ≠ {}"
    then obtain a where "a ∈ A" by auto
    let ?r = "λb. if b ∈ f ` A then the_inv_into A f b else a"
    have "?r ` B ⊆ A"
    proof (rule subsetI)
      fix a'
      assume "a' ∈ ?r ` B"
      then obtain b where "b ∈ B" and "a' = ?r b" by auto
      {
        assume "b ∈ f ` A"
        then obtain a'' where "a'' ∈ A" and "b = f a''" by auto
        from ‹b ∈ f ` A› have "?r b = the_inv_into A f b" by simp
        with ‹a' = ?r b› have "the_inv_into A f b = a'" by simp
        with ‹b = f a''› have "the_inv_into A f (f a'') = a'" by simp
        moreover from this and assms(2) and ‹a'' ∈ A› have "the_inv_into A f (f a'') = a''"
          by (intro the_inv_into_f_f)
        ultimately have "a' = a''" by simp
        with ‹a'' ∈ A› have "a' ∈ A" by simp
      }
      moreover {
        assume "b ∉ f ` A"
        hence "?r b = a" by simp
        with ‹a' = ?r b› have "a' = a" by simp
        with ‹a ∈ A› have "a' ∈ A" by simp
      }
      ultimately show "a' ∈ A" by auto
    qed
    moreover have "id_on (?r ∘ f) A"
    proof (rule id_onI)
      fix a'
      assume "a' ∈ A"
      hence "f a' ∈ f ` A" by simp
      hence "?r (f a') = the_inv_into A f (f a')" by simp
      also from assms(2) and ‹a' ∈ A› have "… = a'" by (intro the_inv_into_f_f)
      finally have "?r (f a') = a'" .
      thus "(?r ∘ f) a' = a'" by simp
    qed
    ultimately have "thesis" by (fact that)
  }
  ultimately show "thesis" by auto
qed

proposition prob_1_5_15:
  assumes "A' = {} ⟹ A = {}" ― ‹This assumption is not specified in the book. However, there
                                   exists a counterexample without it.›
    and "u ` A' ⊆ A"
    and "v ` B ⊆ B'"
  defines Phi: "Φ f ≡ v ∘ f ∘ u"
  obtains "u ` A' = A ⟶ inj_on v B
             ⟶ (∀f ∈ A → B. ∀f' ∈ A → B. ext_eq_on A' (Φ f) (Φ f') ⟶ ext_eq_on A f f')"
    and "inj_on u A' ⟶ v ` B = B' ⟶ (∀f' ∈ A' → B'. ∃f ∈ A → B. ext_eq_on A' (Φ f) f')"
proof -
  have "u ` A' = A ⟶ inj_on v B
          ⟶ (∀f ∈ A → B. ∀f' ∈ A → B. ext_eq_on A' (Φ f) (Φ f') ⟶ ext_eq_on A f f')"
  proof (intro impI)
    assume "u ` A' = A" and "inj_on v B"
    {
      fix f and f'
      assume "f ∈ A → B" and "f' ∈ A → B" and "ext_eq_on A' (Φ f) (Φ f')"
      obtain u' where "right_inv_into A' u u'" by (fact right_inv_into)
      hence "id_on (u ∘ u') (u ` A')" by (fact right_inv_intoD2_pf)
      with ‹u ` A' = A› have "id_on (u ∘ u') A" by simp
      from ‹inj_on v B› obtain v' where "left_inv_into B v v'" by (elim left_inv_into)
      hence *: "id_on (v' ∘ v) B" by (fact left_inv_intoD2_pf)
      from ‹id_on (u ∘ u') A› have "ext_eq_on A (f ∘ (u ∘ u')) f" by (fact thm_1_6_2_a)
      hence "ext_eq_on A (f ∘ u ∘ u') f" by (simp only: comp_assoc)
      moreover have "(f ∘ u ∘ u') ` A ⊆ B"
      proof -
        from ‹id_on (u ∘ u') A› have "(u ∘ u') ` A = A" by (fact id_on_imp_surj_on)
        with ‹f ∈ A → B› show ?thesis by fastforce
      qed
      moreover note *
      ultimately have "ext_eq_on A ((v' ∘ v) ∘ f ∘ u ∘ u') f" by fastforce
      hence "ext_eq_on A (v' ∘ (Φ f) ∘ u') f" unfolding Phi by (simp only: comp_assoc)
      {
        fix a
        assume "a ∈ A"
        moreover from ‹u ` A' = A› and ‹right_inv_into A' u u'› have "u' ` A ⊆ A'"
          by (auto dest: right_inv_intoD1)
        ultimately have "u' a ∈ A'" by auto
        with ‹ext_eq_on A' (Φ f) (Φ f')› have "(Φ f') (u' a) = (Φ f) (u' a)" by auto
        hence "v' ((Φ f') (u' a)) = v' ((Φ f) (u' a))" by simp
        also have "… = v' (v (f (u (u' a))))" by (simp add: Phi)
        also from ‹u ` A' = A› and ‹right_inv_into A' u u'› and ‹a ∈ A› have "… = v' (v (f a))"
          by (simp only: right_inv_intoD2)
        also have "… = f a"
        proof -
          from ‹f ∈ A → B› and ‹a ∈ A› have "f a ∈ B" by auto
          with ‹left_inv_into B v v'› show ?thesis by (fact left_inv_intoD2)
        qed
        finally have "v' ((Φ f') (u' a)) = f a" .
      }
      hence "ext_eq_on A (v' ∘ (Φ f') ∘ u') f" by auto
      hence "ext_eq_on A ((v' ∘ v) ∘ f' ∘ u ∘ u') f" unfolding Phi by (simp only: comp_assoc)
      moreover have "(f' ∘ u ∘ u') ` A ⊆ B"
      proof -
        from ‹id_on (u ∘ u') A› have "(u ∘ u') ` A = A" by (fact id_on_imp_surj_on)
        with ‹f' ∈ A → B› show ?thesis by fastforce
      qed
      moreover note *
      ultimately have "ext_eq_on A (f' ∘ u ∘ u') f" by fastforce
      with ‹id_on (u ∘ u') A› have "ext_eq_on A f' f" by fastforce
      hence "ext_eq_on A f f'" by (fact ext_eq_on_sym)
    }
    thus "∀f ∈ A → B. ∀f' ∈ A → B. ext_eq_on A' (Φ f) (Φ f') ⟶ ext_eq_on A f f'" by simp
  qed
  moreover have "inj_on u A' ⟶ v ` B = B'
                   ⟶ (∀f' ∈ A' → B'. ∃f ∈ A → B. ext_eq_on A' (Φ f) f')"
  proof (intro impI)
    assume "inj_on u A'" and "v ` B = B'"
    {
      fix f'
      assume "f' ∈ A' → B'"
      from ‹inj_on u A'› and assms(1) obtain u'
        where "u' ` A ⊆ A'" and "id_on (u' ∘ u) A'" by (elim thm_1_7_b_a)
      obtain v' where "right_inv_into B v v'" by (fact right_inv_into)
      hence "v' ` v ` B ⊆ B" and "id_on (v ∘ v') (v ` B)"
        by (fact right_inv_intoD1, fact right_inv_intoD2_pf)
      with ‹v ` B = B'› have "v' ` B' ⊆ B" and "id_on (v ∘ v') B'" by simp+
      let ?f = "v' ∘ f' ∘ u'"
      from ‹u' ` A ⊆ A'› and ‹f' ∈ A' → B'› and ‹v' ` B' ⊆ B› have "?f ` A ⊆ B" by fastforce
      moreover have "ext_eq_on A' (Φ ?f) f'"
      proof -
        have "ext_eq_on A' (Φ ?f) (v ∘ v' ∘ f' ∘ u' ∘ u)" unfolding Phi by auto
        also have "ext_eq_on A' … (f' ∘ u' ∘ u)"
        proof -
          from ‹id_on (u' ∘ u) A'› and ‹f' ∈ A' → B'› have "(f' ∘ u' ∘ u) ` A' ⊆ B'" by fastforce
          with ‹id_on (v ∘ v') B'› show ?thesis by fastforce
        qed
        also from ‹id_on (u' ∘ u) A'› have "ext_eq_on A' … f'" by fastforce
        finally show "ext_eq_on A' (Φ ?f) f'" .
      qed
      ultimately have "∃f ∈ A → B. ext_eq_on A' (Φ f) f'" by blast
    }
    thus "∀f' ∈ A' → B'. ∃f ∈ A → B. ext_eq_on A' (Φ f) f'" ..
  qed
  ultimately show "thesis" by (fact that)
qed

proposition prob_1_5_15_ext_a:
  assumes "u ` A' = A"
    and "inj_on v B"
  defines "Φ f ≡ λa'. if a' ∈ A' then v (f (u a')) else undefined"
  shows "inj_on Φ (A →⇩E B)"
proof (rule inj_onI)
  fix f f'
  assume f: "f ∈ A →⇩E B"
    and f': "f' ∈ A →⇩E B"
    and "Φ f = Φ f'"
  {
    fix a
    consider "a ∈ A" | "a ∉ A" by auto
    moreover {
      assume a: "a ∈ A"
      with assms(1) obtain a' where "a' ∈ A'" and "u a' = a" by auto
      with ‹Φ f = Φ f'› and Φ_def have "v (f a) = v (f' a)" by metis
      moreover from f and a have "f a ∈ B" by auto
      moreover from f' and a have "f' a ∈ B" by auto
      moreover note assms(2)
      ultimately have "f a = f' a" by (simp only: inj_onD)
    }
    moreover {
      assume "a ∉ A"
      with f and f' have "f a = f' a" by fastforce
    }
    ultimately have "f a = f' a" by auto
  }
  thus "f = f'" by auto
qed

proposition prob_1_5_15_ext_b:
  fixes A :: "'a set"
    and B :: "'b set"
  assumes "A' = {} ⟹ A = {}"
    and "u ` A' ⊆ A"
    and "inj_on u A'"
    and "v ` B = B'"

    finally have "Φ f a' ∈ B'" .

    fix a'

    with Φ_def have "Φ f a' = undefined" by simp

      }
      moreover {
        assume "a ∉ u ` A'"

      }
      ultimately have "f a ∈ B" by auto
    }

      fix a
      assume "a ∉ A"
      moreover from this and assms(2) have "a ∉ u ` A'" by auto

    }

      }

qed
