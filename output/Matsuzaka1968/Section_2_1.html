definition equipotent :: "'a set ⇒ 'b set ⇒ bool"
  where "equipotent A B ⟷ (∃f. bij_betw f A B)"

lemma equipotentI [intro]:
  assumes "bij_betw f A B"
  shows "equipotent A B"
  using assms unfolding equipotent_def by auto

lemma equipotentE [elim]:
  assumes "equipotent A B"
  obtains f where "bij_betw f A B"
  using assms unfolding equipotent_def by auto

lemma equipotent_empty1 [simp]:

lemma equipotent_empty2 [simp]:

  shows "A = {}"
  using assms by auto

proposition prop_2_1_1 [simp]:
  shows "equipotent A A"

proposition prop_2_1_2:
  assumes "equipotent A B"
  shows "equipotent B A"
proof -
  from assms obtain f where "bij_betw f A B" by auto
  hence "bij_betw (the_inv_into A f) B A" by (fact bij_betw_the_inv_into)
  thus ?thesis by auto
qed

proposition prop_2_1_3 [trans]:
  assumes "equipotent A B"
    and "equipotent B C"
  shows "equipotent A C"
proof -
  from assms obtain f and g where "bij_betw f A B" and "bij_betw g B C" by blast
  hence "bij_betw (g ∘ f) A C" by (rule thm_1_5_c)
  thus ?thesis by auto
qed

proposition ex_2_3_factorization_existence:
  assumes "0 < n"
  obtains i :: nat and j :: nat where "n = 2^i * (2 * j + 1)"
proof -
  have "prime (2 :: nat)" by (fact two_is_prime_nat)
  with assms obtain i and m
    where "¬2 dvd m"
    and "n = m * 2^i" by (blast dest: prime_power_canonical)
  from this(1) obtain j where "m = 2 * j + 1" by (elim oddE)
  with ‹n = m * 2^i› have "n = 2^i * (2 * j + 1)" by simp
  thus "thesis" by (fact that)
qed

proposition ex_2_3_factorization_uniqueness:
  fixes i j i' j' :: nat
  assumes "2^i * (2 * j + 1) = 2^i' * (2 * j' + 1)"
  obtains "i = i'" and "j = j'"
proof -
  from assms have "(2 * j' + 1) * 2^i' = (2 * j + 1) * 2^i" by (simp only: mult.commute)
  moreover have "prime (2 :: nat)" by simp
  moreover have "¬2 dvd (2 * j' + 1)" by simp
  moreover have "¬2 dvd (2 * j + 1)" by simp
  ultimately have "2 * j' + 1 = 2 * j + 1" and "i' = i" using prime_power_cancel2 by blast+
  hence "j = j'" and "i = i'" by simp+
  thus "thesis" by (intro that)
qed

proposition ex_2_3:
  shows "equipotent ((UNIV :: nat set) × (UNIV :: nat set)) (UNIV :: nat set)"
proof -
  let ?f = "λ(i :: nat, j :: nat). 2^i * (2 * j + 1) - 1"
  have "?f ` (UNIV × UNIV) = UNIV"
  proof (rule surj_onI; split_pair)
    fix i j :: nat
    show "2^i * (2 * j + 1) - 1 ∈ UNIV" by simp
  next
    fix n :: nat
    obtain i and j where "n + 1 = 2^i * (2 * j + 1)" using ex_2_3_factorization_existence by auto
    hence "2^i * (2 * j + 1) - 1 = n" by presburger
    thus "n ∈ ?f ` (UNIV × UNIV)" by auto
  qed
  moreover have "inj_on ?f (UNIV × UNIV)"
  proof (rule inj_onI, split_pair)
    thm inj_onI
    fix i j i' j' :: nat
    assume "2^i * (2 * j + 1) - 1 = 2^i' * (2 * j' + 1) - 1"
    moreover have "0 < 2^i * (2 * j + 1)" and "0 < 2^i' * (2 * j' + 1)" by simp+
    ultimately have "2^i * (2 * j + 1) = 2^i' * (2 * j' + 1)" by linarith
    hence "i = i'" and "j = j'" using ex_2_3_factorization_uniqueness by blast+
    thus "(i, j) = (i', j')" by simp
  qed
  ultimately have "bij_betw ?f (UNIV × UNIV) UNIV" by (intro bij_betw_imageI)
  thus "equipotent ((UNIV :: nat set) × (UNIV :: nat set)) (UNIV :: nat set)" by auto
qed

proposition ex_2_4:
  assumes "(a :: real) < (b :: real)"
    and "(c :: real) < (d :: real)"
  shows "equipotent {a .. b} {c .. d}"
proof -
  define f where "f x ≡ (d - c) * (x - a) / (b - a) + c" for x
  have "bij_betw f {a .. b} {c .. d}"
  proof (rule bij_betw_imageI')
    fix x and x'
    assume "x ∈ {a .. b}"
      and "x' ∈ {a .. b}"
      and "f x = f x'"
    with f_def have "(d - c) * (x - a) / (b - a) = (d - c) * (x' - a) / (b - a)" by simp
    with assms(1) have "(d - c) * (x - a) = (d - c) * (x' - a)" by simp
    with assms(2) show "x = x'" by simp
  next
    fix x
    assume x: "x ∈ {a .. b}"
    have "c ≤ f x"
    proof -
      from assms(2) have "0 ≤ d - c" by simp
      moreover from x have "0 ≤ x - a" by simp
      moreover from assms(1) have "0 ≤ b - a" by simp
      ultimately have "0 ≤ (d - c) * (x - a) / (b - a)" by simp
      with f_def show ?thesis by simp
    qed
    moreover have "f x ≤ d"
    proof -
      from x and assms(1) have "(x - a) / (b - a) ≤ 1" by simp
      moreover from assms(2) have "0 ≤ d - c" by simp
      ultimately have "(d - c) * ((x - a) / (b - a)) ≤ d - c" by (fact mult_left_le)
      thus ?thesis unfolding f_def by simp
    qed
    ultimately show "f x ∈ {c .. d}" by simp
  next
    fix y
    assume y: "y ∈ {c .. d}"
    define x where "x ≡ (y - c) * (b - a) / (d - c) + a"
    with f_def have "f x = y"
    proof -
      have "f x = (d - c) * (((y - c) * (b - a) / (d - c) + a) - a) / (b - a) + c"
        unfolding f_def and x_def ..
      also from assms have "… = y" by simp
      finally show ?thesis .
    qed
    moreover have "x ∈ {a .. b}"
    proof -
      have "a ≤ x"
      proof -
        from y have "0 ≤ y - c" by simp
        moreover from assms(1) have "0 ≤ b - a" by simp
        moreover from assms(2) have "0 ≤ d - c" by simp
        ultimately have "0 ≤ (y - c) * (b - a) / (d - c)" by simp
        thus ?thesis unfolding x_def by simp
      qed
      moreover have "x ≤ b"
      proof -
        from y and assms(2) have "(y - c) / (d - c) ≤ 1" by simp
        moreover from assms(1) have "0 ≤ (b - a)" by simp
        ultimately have "(b - a) * ((y - c) / (d - c)) ≤ b - a" by (fact mult_left_le)
        hence "(y - c) * (b - a) / (d - c) + a ≤ b" by argo
        thus ?thesis unfolding x_def by simp
      qed
      ultimately show ?thesis by simp
    qed
    ultimately show "y ∈ f ` {a .. b}" by auto
  qed
  thus ?thesis by (fact equipotentI)
qed

lemma non_negative_quadratic_discriminant_implies_real_root:
  assumes "a ≠ 0"
    and "discrim a b c ≥ 0"
  obtains x where "a * x⇧2 + b * x + c = 0"
    and "x = (-b + sqrt(discrim a b c)) / (2 * a)"
  using assms and discriminant_iff by blast

proposition ex_2_5:
  shows "equipotent {-(1 :: real) <..< (1 :: real)} (UNIV :: real set)"
proof -
  define f where "f x ≡ x / (1 - x⇧2)" for x :: real
  have "bij_betw f {-(1 ::real) <..< (1 :: real)} (UNIV :: real set)"
  proof (rule bij_betw_imageI')
    fix x and x'
    assume x: "x ∈ {-1 <..< 1}"
      and x': "x' ∈ {-1 <..< 1}"
      and "f x = f x'"
    from this(3) have "x / (1 - x⇧2) = x' / (1 - x'⇧2)" unfolding f_def .
    hence "(1 - x⇧2) * (1 - x'⇧2) * (x / (1 - x⇧2)) = (1 - x⇧2) * (1 - x'⇧2) * (x' / (1 - x'⇧2))" by simp
    moreover have "1 - x⇧2 ≠ 0"
    proof (rule notI)
      assume "1 - x⇧2 = 0"
      hence "x = 1 ∨ x = -1" by algebra
      with x show "False" by simp
    qed
    moreover have "1 - x'⇧2 ≠ 0"
    proof (rule notI)
      assume "1 - x'⇧2 = 0"
      hence "x' = 1 ∨ x' = -1" by algebra
      with x' show "False" by simp
    qed
    ultimately have "(1 - x'⇧2) * x = (1 - x⇧2) * x'" by auto
    hence "x - x * x'⇧2 - x' + x' * x⇧2 = 0" by argo
    hence *: "x - x' + x * x' * (x - x') = 0" by algebra
    {
      assume "x ≠ x'"
      with * have "1 + x * x' = 0" by algebra
      moreover from x and x' have "-1 < x * x'"
      proof -
        consider (A) "x * x' ≥ 0"
            | (B) "x > 0"
            | (C) "x' > 0"
          by (metis less_eq_real_def linorder_neqE_linordered_idom zero_le_mult_iff)
        thus ?thesis
        proof cases
          case A
          thus ?thesis by simp
        next
          case B
          moreover from x' have "-1 < x'" by simp
          ultimately have "x * (-1) < x * x'" by (blast dest: mult_strict_left_mono)
          hence "-x < x * x'" by simp
          moreover from x have "-1 < -x" by simp
          ultimately show ?thesis by simp
        next
          case C
          moreover from x have "-1 < x" by simp
          ultimately have "(-1) * x' < x * x'" by (blast dest: mult_strict_right_mono)
          hence "-x' < x * x'" by simp
          moreover from x' have "-1 < -x'" by simp
          ultimately show ?thesis by simp
        qed
      qed
      ultimately have "False" by simp
    }
    thus "x = x'" by (fact ccontr)
  next
    fix x :: real
    show "f x ∈ UNIV" by simp
  next
    fix y :: real
    {
      assume y: "y = 0"
      with f_def have  "f 0 = y" by simp
      hence "y ∈ f ` {-1 <..< 1}" by auto
    }
    moreover {
      assume "y ≠ 0"
      have discrim: "discrim y 1 (-y) ≥ 0" unfolding discrim_def by simp
      from ‹y ≠ 0› and discrim obtain x where x_root: "y * x⇧2 + 1 * x + (-y) = 0"
        and x: "x = (-1 + sqrt(discrim y 1 (-y))) / (2 * y)"
        by (elim non_negative_quadratic_discriminant_implies_real_root)
      have "f x = y"
      proof -
        from x_root have "(1 - x⇧2) * y = x" by argo
        moreover have "1 - x⇧2 ≠ 0"
        proof (rule notI)
          assume "1 - x⇧2 = 0"
          moreover from this and x_root have "x = 0" by simp
          ultimately show "False" by simp
        qed
        ultimately have "y = x / (1 - x⇧2)" by (metis nonzero_mult_div_cancel_left)
        thus ?thesis unfolding f_def by simp
      qed
      moreover have "x ∈ {-1 <..< 1}"
      proof -
        {
          assume "y > 0"
          have "0 < x"
          proof -
            from ‹y > 0› have "0 < -1 + sqrt(1 + 4 * y * y)" by simp
            moreover from ‹y > 0› have "0 < 2 * y" by simp
            ultimately have "0 < (-1 + sqrt(1 + 4 * y * y)) / (2 * y)" by simp
            with x show "0 < x" unfolding discrim_def by simp
          qed
          moreover have "x < 1"
          proof -
            from ‹y > 0› have "1 + 4 * y * y < (1 + 2 * y) * (1 + 2 * y)" by argo
            hence "sqrt(1 + 4 * y * y) < sqrt((1 + 2 * y) * (1 + 2 * y))"
              by (fact real_sqrt_less_mono)
            moreover from ‹y > 0› have "0 < 1 + 2 * y" by simp
            ultimately have "sqrt(1 + 4 * y * y) < 1 + 2 * y" by simp
            hence "-1 + sqrt(1 + 4 * y * y) < 2 * y" by simp
            moreover from ‹y > 0› have "0 < 2 * y" by simp
            ultimately have "(-1 + sqrt(1 + 4 * y * y)) / (2 * y) < 1" by simp
            with x show "x < 1" unfolding discrim_def by simp
          qed
          ultimately have "x ∈ {-1 <..< 1}" by simp
        }
        moreover {
          assume "y < 0"
          have "-1 < x"
          proof -
            from ‹y < 0› have "0 < -4 * y" by simp
            hence "1 + 4 * y * y < (1 - 2 * y) * (1 - 2 * y)" by argo
            hence "sqrt(1 + 4 * y * y) < sqrt((1 - 2 * y) * (1 - 2 * y))"
              by (fact real_sqrt_less_mono)
            moreover from ‹y < 0› have "0 < 1 - 2 * y" by simp
            ultimately have "sqrt(1 + 4 * y * y) < 1 - 2 * y" by simp
            hence "2 * y < 1 - sqrt(1 + 4 * y * y)" by simp
            moreover from ‹y < 0› have "2 * y < 0" by simp
            ultimately have "(1 - sqrt(1 + 4 * y * y)) / (2 * y) < 1" by simp
            hence "-1 < (-1 + sqrt(1 + 4 * y * y)) / (2 * y)" by argo
            with x show ?thesis unfolding discrim_def by simp
          qed
          moreover have "x < 0"
          proof -
            from ‹y < 0› have "y ≠ 0" by simp
            hence "0 < y * y" using not_real_square_gt_zero by blast
            hence "0 < -1 + sqrt(1 + 4 * y * y)" by simp
            moreover from ‹y < 0› have "2 * y < 0" by simp
            ultimately have "(-1 + sqrt(1 + 4 * y * y)) / (2 * y) < 0" by (fact divide_pos_neg)
            with x show ?thesis unfolding discrim_def by simp
          qed
          ultimately have "x ∈ {-1 <..< 1}" by simp
        }
        moreover note ‹y ≠ 0›
        ultimately show "x ∈ {-1 <..< 1}" by argo
      qed
      ultimately have "y ∈ f ` {-1 <..< 1}" by auto
    }
    ultimately show "y ∈ f ` {-1 <..< 1}" by auto
  qed
  thus ?thesis by auto
qed

fun bernstein_seq :: "'a set ⇒ 'b set ⇒ ('a ⇒ 'b) ⇒ ('b ⇒ 'a) ⇒ nat ⇒ 'a set"
  and bernstein_seq' :: "'a set ⇒ 'b set ⇒ ('a ⇒ 'b) ⇒ ('b ⇒ 'a) ⇒ nat ⇒ 'b set" where
    "bernstein_seq' A B f g 0 = B - f ` A"
  | "bernstein_seq A B f g n = g ` bernstein_seq' A B f g n"
  | "bernstein_seq' A B f g (Suc n) = f ` bernstein_seq A B f g n"

lemma bernstein_seq_subset:
  assumes "f ` A ⊆ B"
    and "g ` B ⊆ A"
  shows "bernstein_seq A B f g n ⊆ A"
proof (induct n)
  case 0
  have "bernstein_seq A B f g 0 = g ` (B - f ` A)" by simp
  also from assms have "… ⊆ A" by auto
  finally show "?case" .
next
  case (Suc n)
  have "bernstein_seq A B f g (Suc n) = g ` f ` bernstein_seq A B f g n" by simp
  with assms and Suc.hyps show "?case" by fastforce
qed

lemma bernstein_seq'_subset:
  assumes "f ` A ⊆ B"
    and "g ` B ⊆ A"
  shows "bernstein_seq' A B f g n ⊆ B"
proof (induct n)
  case 0
  have "bernstein_seq' A B f g 0 = B - f ` A" by simp
  also have "… ⊆ B" by auto
  finally show "?case" .
next
  case (Suc n)
  have "bernstein_seq' A B f g (Suc n) = f ` g ` bernstein_seq' A B f g n" by simp
  with assms and Suc.hyps show "?case" by fastforce
qed

lemma zero_Un_UNION_Suc_eq:
  shows "A 0 ∪ (⋃n. A (Suc n)) = (⋃n. A n)"
proof -
  have "A 0 ∪ (⋃n. A (Suc n)) = (⋃n ∈ {0}. A n) ∪(⋃n ∈ {Suc n | n. True}. A n)" by auto
  also have "… = (⋃n ∈ {0} ∪ {Suc n | n. True}. A n)" by simp
  also have "… = (⋃n. A n)"
  proof -
    {
      fix n
      have "n ∈ {0} ∪ {Suc n | n. True}"
      proof (induct n)
        case 0
        show "?case" by simp
      next
        case (Suc n)
        show "?case" by simp
      qed
    }
    hence "{0} ∪ {Suc n | n. True} = UNIV" by blast
    thus ?thesis by simp
  qed
  finally show ?thesis .
qed

theorem thm_2_2:
  assumes "f ` A ⊆ B"
    and "inj_on f A"
    and "g ` B ⊆ A"
    and "inj_on g B"
  shows "equipotent A B"
proof -
  let ?A = "⋃n. bernstein_seq A B f g n"
  let ?A' = "A - ?A"
  let ?B = "⋃n. bernstein_seq' A B f g n"
  let ?B' = "B - ?B"
  {
    fix n
    from assms(1,3) have "bernstein_seq A B f g n ⊆ A" by (rule bernstein_seq_subset)
  }
  hence "?A ⊆ A" by auto
  hence "?A' ⊆ A" by auto
  {
    fix n
    from assms(1,3) have "bernstein_seq' A B f g n ⊆ B" by (rule bernstein_seq'_subset)
  }
  hence "?B ⊆ B" by auto
  hence "?B' ⊆ B" by auto
  have "f ` ?A' = ?B'"
  proof -
    from ‹?A ⊆ A› and assms(2) have "f ` ?A' = f ` A - f ` ?A" by (intro prob_1_4_5_c)
    also have "… = f ` A - (⋃n. f ` bernstein_seq A B f g n)" by blast
    also have "… = f ` A - (⋃n. bernstein_seq' A B f g (Suc n))" by simp
    also from assms(1) have "… = B - (B - f ` A) - (⋃n. bernstein_seq' A B f g (Suc n))" by auto
    also have "… = B - bernstein_seq' A B f g 0 - (⋃n. bernstein_seq' A B f g (Suc n))" by simp
    also have "… = B - (bernstein_seq' A B f g 0 ∪ (⋃n. bernstein_seq' A B f g (Suc n)))" by auto
    also have "… = ?B'"
      by (simp only: zero_Un_UNION_Suc_eq[where A = "λn. bernstein_seq' A B f g n"])
    finally show ?thesis .
  qed
  moreover from assms(2) have "inj_on f ?A'" by (fact inj_on_diff)
  ultimately have "bij_betw f ?A' ?B'" by (intro bij_betw_imageI)
  have "g ` ?B = ?A"
  proof (rule surj_onI)
    fix b
    assume "b ∈ ?B"
    then obtain n where "b ∈ bernstein_seq' A B f g n" by fast
    hence "g b ∈ bernstein_seq A B f g n" by simp
    thus "g b ∈ ?A" by auto
  next
    fix a
    assume "a ∈ ?A"
    then obtain n where "a ∈ bernstein_seq A B f g n" by blast
    hence "a ∈ g ` bernstein_seq' A B f g n" by simp
    then obtain b where "b ∈ bernstein_seq' A B f g n" and "a = g b" by auto
    from this(1) have "b ∈ ?B" by auto
    with ‹a = g b› show "a ∈ g ` ?B" by simp
  qed
  moreover from ‹?B ⊆ B› and assms(4) have "inj_on g ?B" by (elim inj_on_subset)
  ultimately have "bij_betw g ?B ?A" by (intro bij_betw_imageI)
  then obtain f' where "bij_betw f' ?A ?B" using bij_betw_inv by blast
  hence "f' ` ?A = ?B" by (fact bij_betw_imp_surj_on)
  let ?F = "λa. if a ∈ ?A' then f a else f' a"
  have "?F ` A = B"
  proof (rule surj_onI)
    fix a
    assume "a ∈ A"
    {
      assume "a ∈ ?A'"
      hence "?F a = f a" by simp
      with ‹a ∈ A› and assms(1) have "?F a ∈ B" by fast
    }
    moreover {
      assume "a ∉ ?A'"
      hence "?F a = f' a" by argo
      also have "… ∈ ?B"
      proof -
        from ‹a ∈ A› and ‹a ∉ ?A'› have "a ∈ ?A" by simp
        moreover note ‹f' ` ?A = ?B›
        ultimately show "f' a ∈ ?B" by blast
      qed
      also note ‹… ⊆ B›
      finally have "?F a ∈ B" .
    }
    ultimately show "?F a ∈ B" by blast
  next
    fix b
    assume "b ∈ B"
    {
      assume "b ∈ ?B"
      with ‹f' ` ?A = ?B› obtain a where "a ∈ ?A" and "b = f' a" by blast
      hence "?F a = b" by simp
      moreover from ‹a ∈ ?A› and ‹?A ⊆ A› have "a ∈ A" ..
      ultimately have "b ∈ ?F ` A" by blast
    }
    moreover {
      assume "b ∉ ?B"
      with ‹b ∈ B› have "b ∈ ?B'" by simp
      with ‹f ` ?A' = ?B'› have "b ∈ f ` ?A'" by simp
      then obtain a where "a ∈ ?A'" and "b = f a" by blast
      hence "?F a = b" by argo
      moreover from ‹a ∈ ?A'› and ‹?A' ⊆ A› have "a ∈ A" ..
      ultimately have "b ∈ ?F ` A" by blast
    }
    ultimately show "b ∈ ?F ` A" by blast
  qed
  moreover have "inj_on ?F A"
  proof (rule inj_onI)
    fix a and a'
    assume "a ∈ A" and
      "a' ∈ A" and
      "?F a = ?F a'"
    consider (A) "a ∈ ?A'" and "a' ∈ ?A'" |
      (B) "a ∈ ?A'" and "a' ∉ ?A'" |
      (C) "a ∉ ?A'" and "a' ∈ ?A'" |
      (D) "a ∉ ?A'" and "a' ∉ ?A'" by blast
    thus "a = a'"
    proof cases
      case A
      with ‹?F a = ?F a'› have "f a = f a'" by argo
      with A and ‹inj_on f ?A'› show ?thesis by (elim inj_onD)
    next
      case B
      from B(2) and ‹a' ∈ A› have "a' ∈ ?A" by blast
      from B and ‹?F a = ?F a'› have "f a = f' a'" by argo
      moreover from ‹a ∈ ?A'› and ‹f ` ?A' = ?B'› have "f a ∈ ?B'" by blast
      moreover from ‹a' ∈ ?A› and ‹f' ` ?A = ?B› have "f' a' ∈ ?B" by blast
      ultimately have "False" by simp
      thus ?thesis ..
    next
      case C
      from C(1) and ‹a ∈ A› have "a ∈ ?A" by simp
      from C and ‹?F a = ?F a'› have "f' a = f a'" by argo
      moreover from ‹a ∈ ?A› and ‹f' ` ?A = ?B› have "f' a ∈ ?B" by blast
      moreover from ‹a' ∈ ?A'› and ‹f ` ?A' = ?B'› have "f a' ∈ ?B'" by blast
      ultimately have "False" by simp
      thus ?thesis ..
    next
      case D
      from D and ‹?F a = ?F a'› have "f' a = f' a'" by argo
      moreover from D and ‹a ∈ A› and ‹a' ∈ A› have "a ∈ ?A" and "a' ∈ ?A" by simp+
      moreover from ‹bij_betw f' ?A ?B› have "inj_on f' ?A" by auto
      ultimately show ?thesis by (elim inj_onD)
    qed
  qed
  ultimately have "bij_betw ?F A B" by (intro bij_betw_imageI)
  thus ?thesis by auto
qed

lemma ex_2_5':
  assumes "(a :: real) < (b :: real)"
    and "(c :: real) < (d :: real)"
  obtains f where "f ` {a <..< b} ⊆ {c <..< d}"
    and "inj_on f {a <..< b}"
proof -
  define f where "f x ≡ (d - c) * (x - a) / (b - a) + c" for x
  have "f ` {a <..< b} ⊆ {c <..< d}"
  proof (rule image_subsetI)
    fix x
    assume x: "x ∈ {a <..< b}"
    have "c < f x"
    proof -
      from assms(1) have "0 < b - a" by simp
      moreover from assms(2) have "0 < d - c" by simp
      moreover from x have "0 < x - a" by simp
      ultimately show ?thesis unfolding f_def by simp
    qed
    moreover have "f x < d"
    proof -
      from assms(1) have "0 < b - a" by simp
      moreover from x have "0 < x - a" by simp
      moreover from x have "x - a < b - a" by simp
      ultimately have "(x - a) / (b - a) < 1" by simp
      moreover from assms(2) have "0 < d - c" by simp
      ultimately have "(d - c) * ((x - a) / (b - a)) < (d - c) * 1" by (fact mult_strict_left_mono)
      thus ?thesis unfolding f_def by simp
    qed
    ultimately show "f x ∈ {c <..< d}" by simp
  qed
  moreover have "inj_on f {a <..< b}"
  proof (rule inj_onI)
    fix x and x'
    assume "f x = f x'"
    hence "(d - c) * (x - a) / (b - a) = (d - c) * (x' - a) / (b - a)" unfolding f_def by simp
    moreover from assms(1) have "b - a ≠ 0" by simp
    ultimately have "(d - c) * (x - a) = (d - c) * (x' - a)" by simp
    moreover from assms(2) have "d - c ≠ 0" by simp
    ultimately show "x = x'" by simp
  qed
  ultimately show thesis by (fact that)
qed

lemma surj_on_from_subset_imp_surj_on:
  assumes "f ` A' = B"
    and "A' ⊆ A"
    and "A' = {} ⟹ A = {}"
  obtains f' where "f' ` A = B"
proof -
  {
    assume "A' = {}"
    with assms(1,3) have "A = {}" and "B = {}" by simp+
    hence "∃f'. f' ` A = B" by simp
  }
  moreover {
    assume "A' ≠ {}"
    with assms(1) have "B ≠ {}" by auto
    then obtain b where "b ∈ B" by auto
    let ?f' = "λa. if a ∈ A' then f a else b"
    have "?f' ` A = B"
    proof (rule surj_onI)
      fix a
      assume "a ∈ A"
      {
        assume "a ∈ A'"
        with assms(1) have "?f' a ∈ B" by auto
      }
      moreover {
        assume "a ∉ A'"
        with ‹b ∈ B› have "?f' a ∈ B" by simp
      }
      ultimately show "?f' a ∈ B" by simp
    next
      fix b'
      assume "b' ∈ B"
      with assms(1) obtain a where "a ∈ A'" and "f a = b'" by blast
      hence "?f' a = b'" by simp
      moreover from ‹a ∈ A'› and assms(2) have "a ∈ A" ..
      ultimately show "b' ∈ ?f' ` A" by auto
    qed
    hence "∃f'. f' ` A = B" by blast
  }
  ultimately show "thesis" using that by auto
qed

theorem thm_2_2''':
  assumes "B⇩1 ⊆ B"
    and "equipotent A B⇩1"
    and "A⇩1 ⊆ A"
    and "equipotent B A⇩1"
  shows "equipotent A B"
proof -
  from assms(2) obtain f where "bij_betw f A B⇩1" by auto
  then obtain g where "bij_betw g B⇩1 A" by (auto dest: bij_betw_inv)
  hence "g ` B⇩1 = A" by auto
  moreover have "B⇩1 = {} ⟹ B = {}"
  proof -
    assume "B⇩1 = {}"
    with assms(2) have "A = {}" by simp
    with assms(3) have "A⇩1 = {}" by simp
    with assms(4) show "B = {}" by simp
  qed
  moreover note assms(1)
  ultimately obtain g' where "g' ` B = A" using surj_on_from_subset_imp_surj_on by blast
  from assms(4) obtain g'' where "bij_betw g'' B A⇩1" by auto
  then obtain f' where "bij_betw f' A⇩1 B" by (auto dest: bij_betw_inv)
  hence "f' ` A⇩1 = B" by auto
  moreover have "A⇩1 = {} ⟹ A = {}"
  proof -
    assume "A⇩1 = {}"
    with assms(4) have "B = {}" by simp
    with assms(1) have "B⇩1 = {}" by simp
    with assms(2) show "A = {}" by simp
  qed
  moreover note assms(3)
  ultimately obtain f'' where "f'' ` A = B" by (auto intro: surj_on_from_subset_imp_surj_on)
  with ‹g' ` B = A› show "equipotent A B" by (intro thm_2_2'')
qed

  }
  then obtain b where "b ∈ (Π⇩E l ∈ Λ. A l - {a l})" by (elim AC_E_ext)
  let ?f = "λl. λl'. if l' = l then a l' else b l'"
  have "?f ` Λ ⊆ (Π⇩E l ∈ Λ. A l)"
  proof (rule subsetI)
    fix c
    assume "c ∈ ?f ` Λ"
    then obtain l where "l ∈ Λ" and "c = ?f l" by auto
    {
      fix l'
      assume "l' ∈ Λ"
      from ‹c = ?f l› have "c l' = ?f l l'" by simp
      {
        assume "l' = l"
        hence "?f l l' = a l'" by simp
        also from this and ‹a ∈ (Π⇩E l ∈ Λ. A l)› and ‹l' ∈ Λ› have "a l' ∈ A l'" by auto
        finally have "?f l l' ∈ A l'" .
      }
      moreover {
        assume "l' ≠ l"
        hence "?f l l' = b l'" by simp
        also from this and ‹b ∈ (Π⇩E l ∈ Λ. A l - {a l})› and ‹l' ∈ Λ› have "b l' ∈ A l'" by auto
        finally have "?f l l' ∈ A l'" .
      }
      ultimately have "?f l l' ∈ A l'" by simp
      with ‹c l' = ?f l l'› have "c l' ∈ A l'" by simp
    }
    moreover {
      fix l'
      assume "l' ∉ Λ"
      with ‹l ∈ Λ› have "l' ≠ l" by auto
      from ‹c = ?f l› have "c l' = ?f l l'"  by simp
      also from ‹l' ≠ l› have "?f l l' = b l'" by simp
      also from ‹l' ∉ Λ› and ‹b ∈ (Π⇩E l ∈ Λ. A l - {a l})› have "b l' = undefined" by auto
      finally have "c l' = undefined" .
    }
    ultimately show "c ∈ (Π⇩E l ∈ Λ. A l)" by (intro PiE_I)
  qed
  moreover have "inj_on ?f Λ"
  proof (rule inj_onI)
    fix l and l'
    assume "l ∈ Λ" and
      "l' ∈ Λ" and
      "?f l = ?f l'"
    {
      assume "l ≠ l'"
      hence "?f l' l = b l" by simp
      moreover have "?f l l = a l" by simp
      moreover note ‹?f l = ?f l'›
      ultimately have "a l = b l" by metis
      also from ‹l ∈ Λ› and ‹b ∈ (Π⇩E l ∈ Λ. A l - {a l})› have "b l ∈ A l - {a l}" by blast
      finally have "a l ∈ A l - {a l}" .
      hence "False" by simp
    }
    thus "l = l'" by auto
  qed
  ultimately show ?thesis by auto
qed

proposition prob_2_1_7:
  assumes "f ` A = B"
  obtains R where "equiv A R" and "equipotent B (A // R)"
proof -
  from assms have "f ` A ⊆ B" by simp
  then obtain g where "bij_betw g (A // (equiv_kernel_on f A)) (f ` A)"
    by (fastforce elim: prop_1_6_4)
  with assms have "bij_betw g (A // (equiv_kernel_on f A)) B" by blast
  hence "equipotent (A // equiv_kernel_on f A) B" by auto
  hence "equipotent B (A // equiv_kernel_on f A)" by (fact prop_2_1_2)
  moreover have "equiv A (equiv_kernel_on f A)" by (fact equiv_equiv_kernel_on)
  ultimately show "thesis" by (intro that)
qed
