fun dc_seq :: "'a set ⇒ ('a set ⇒ 'a) ⇒ nat ⇒ 'a set" where
"dc_seq M a 0 = {}" |
"dc_seq M a (Suc n) = dc_seq M a n ∪ {a (M - dc_seq M a n)}"

lemma finite_dc_seq:
  shows "finite (dc_seq M a n)"
proof (induct n)
  case 0
  show "?case" by simp
next
  case (Suc n)
  from Suc.hyps show "?case" by simp
qed

lemma strict_dec_induct [consumes 1, case_names base step]:
  assumes "i < j"
    and "P (Suc i)"
    and "⋀j. i < j ⟹ P j ⟹ P (Suc j)"
  shows "P j"
proof -
  let ?P = "λn. P (n + i + 1)"
  have "?P n" for n
  proof (induct n)
    case 0
    from assms(2) show "?P 0" by simp
  next
    case (Suc n)
    note ‹?P n›
    from Suc.hyps have "P (n + i + 1)" by simp
    moreover have "i < n + i + 1" by simp
    ultimately have "P (Suc (n + i + 1))" by (intro assms(3))
    thus "?P (Suc n)" by simp
  qed
  from assms(1) obtain k where "j = Suc (i + k)" by (auto dest: less_imp_Suc_add)
  hence "j = k + i + 1" by simp
  with ‹?P k› show "P j" by simp
qed

theorem thm_2_4:
  assumes "infinite M"
  obtains A where "A ⊆ M" and "|A| =o ℵ⇩0"
proof -
  let ?𝔐 = "Pow M - {{}}"
  {
    fix A
    assume "A ∈ ?𝔐"
    hence "A ≠ {}" by simp
  }
  then obtain a where "a ∈ (Π A ∈ ?𝔐. A)" by (elim AC_E)
  have *: "dc_seq M a n ⊂ M" for n
  proof (induct n)
    case 0
    from assms have "M ≠ {}" by auto
    thus "?case" by auto
  next
    case (Suc n)
    from Suc.hyps have "M - dc_seq M a n ≠ {}" by auto
    moreover have "M - dc_seq M a n ⊆ M" by auto
    ultimately have "M - dc_seq M a n ∈ ?𝔐" by simp
    with ‹a ∈ (Π A ∈ ?𝔐. A)› have "a (M - dc_seq M a n) ∈ M - dc_seq M a n" by (fact Pi_mem)
    also have "M - dc_seq M a n ⊆ M" by auto
    finally have "a (M - dc_seq M a n) ∈ M" .
    moreover note Suc.hyps
    ultimately have "dc_seq M a (Suc n) ⊆ M" by simp
    moreover from assms and finite_dc_seq have "dc_seq M a (Suc n) ≠ M" by metis
    ultimately show "?case" by auto
  qed
  hence "M - dc_seq M a n ∈ ?𝔐" for n by auto
  let ?A = "⋃n. dc_seq M a n"
  let ?f = "λn. a (M - dc_seq M a n)"
  have **: "?f n ∈ M - dc_seq M a n" for n
    using ‹a ∈ (Π A ∈ ?𝔐. A)› and ‹M - dc_seq M a n ∈ ?𝔐› by (fact Pi_mem)
  have ***: "n < n' ⟹ ?f n ∈ dc_seq M a n'" for n and n'
  proof (induct n' rule: strict_dec_induct)
    case base
    show "a (M - dc_seq M a n) ∈ dc_seq M a (Suc n)" by simp
  next
    case (step n')
    from step.hyps(2) show "a (M - dc_seq M a n) ∈ dc_seq M a (Suc n')" by simp
  qed
  have "bij_betw ?f (UNIV :: nat set) ?A"
  proof (rule bij_betw_imageI)
    show "inj_on ?f (UNIV :: nat set)"
    proof (rule inj_onI)
      fix n and n'
      assume "?f n = ?f n'"
      {
        assume "n < n'"
        with *** have "?f n ∈ dc_seq M a n'" by simp
        moreover from ** have "?f n' ∉ dc_seq M a n'" by simp
        moreover note ‹?f n = ?f n'›
        ultimately have "False" by simp
      }
      moreover {
        assume "n' < n"
        with *** have "?f n' ∈ dc_seq M a n" by simp
        moreover from ** have "?f n ∉ dc_seq M a n" by simp
        moreover note ‹?f n = ?f n'›
        ultimately have "False" by simp
      }
      ultimately show "n = n'" by fastforce
    qed
    show "?f ` UNIV = ?A"
    proof (rule surj_onI)
      fix n
      have "?f n ∈ dc_seq M a n ∪ {?f n}" by simp
      hence "?f n ∈ dc_seq M a (Suc n)" by simp
      thus "?f n ∈ ?A" by blast
    next
      fix x
      assume "x ∈ ?A"
      then obtain n where "x ∈ dc_seq M a n" by auto
      {
        assume "∀n. ?f n ≠ x"
        have "x ∉ dc_seq M a n'" for n'
        proof (induct n')
          case 0
          show "?case" by simp
        next
          case (Suc n)
          from Suc.hyps and ‹∀n. ?f n ≠ x› show "?case" by auto
        qed
        with ‹x ∈ dc_seq M a n› have "False" by simp
      }
      thus "x ∈ ?f ` UNIV" by auto
    qed
  qed
  hence "ℵ⇩0 =o |?A|" by auto
  hence "|?A| =o ℵ⇩0" by (fact card_eq_sym)
  moreover from * have "?A ⊆ M" by auto
  ultimately show "thesis" by (intro that)
qed

theorem thm_2_5_1_a:
  assumes "|A| ≤o ℵ⇩0"
    and "|B| ≤o ℵ⇩0"
  shows "|A × B| ≤o ℵ⇩0"
proof -
  from assms(1) obtain f where "f ` A ⊆ (UNIV :: nat set)" and "inj_on f A" by auto
  from assms(2) obtain g where "g ` B ⊆ (UNIV :: nat set)" and "inj_on g B" by auto
  from ‹f ` A ⊆ UNIV› and ‹g ` B ⊆ UNIV› have "map_prod f g ` (A × B) ⊆ UNIV × UNIV" by simp
  moreover from ‹inj_on f A› and ‹inj_on g B› have "inj_on (map_prod f g) (A × B)"
    by (rule map_prod_inj_on)
  ultimately have "|A × B| ≤o |(UNIV :: nat set) × (UNIV :: nat set)|" by auto
  moreover note nat_Times_nat_card_eq_aleph_zero
  ultimately show ?thesis by (fact card_leq_card_eq_trans)
qed

lemma thm_2_5_1_b_a:
  assumes "|A| =o ℵ⇩0"
    and "|B| ≤o ℵ⇩0"
    and "B ≠ {}"
  shows "|A × B| =o ℵ⇩0"
proof -
  from assms(1) have "ℵ⇩0 =o |A|" by (fact card_eq_sym)
  then obtain f where "bij_betw f (UNIV :: nat set) A" by auto
  from assms(3) obtain b where "b ∈ B" by auto
  let ?g = "λn. (f n, b)"
  have "?g ` UNIV ⊆ A × B"
  proof (rule image_subsetI)
    fix n
    from ‹bij_betw f UNIV A› have "f n ∈ A" by auto
    with ‹b ∈ B› show "?g n ∈ A × B" by simp
  qed
  moreover have "inj_on ?g UNIV"
  proof (rule inj_onI)
    fix m n
    assume "(f m, b) = (f n, b)"
    hence "f m = f n" by simp
    moreover from ‹bij_betw f UNIV A› have "inj_on f UNIV" by auto
    ultimately show "m = n" by (auto dest: injD)
  qed
  ultimately have "ℵ⇩0 ≤o |A × B|" by auto
  from assms(1) have "|A| ≤o ℵ⇩0" by blast
  with assms(2) have "|A × B| ≤o ℵ⇩0" by (intro thm_2_5_1_a)
  with ‹ℵ⇩0 ≤o |A × B|› show ?thesis by (intro thm_2_3_2)
qed

lemma Times_card_commute:
  fixes A :: "'a set"
  shows "|A × B| =o |B × A|"
proof -
  have "prod.swap ∈ A × B → B × A" by auto
  moreover have "prod.swap ∈ B × A → A × B" by auto
  moreover {
    fix x
    assume "x ∈ A × B"
    have "prod.swap (prod.swap x) = x" by simp
  }
  moreover {
    fix x
    assume "x ∈ B × A"
    have "prod.swap (prod.swap x) = x" by simp
  }
  ultimately have "bij_betw prod.swap (A × B) (B × A)" by (fact bij_betwI)
  thus ?thesis by auto
qed

lemma card_leq_imp_surj_on:
  assumes "|A| ≤o |B|"
    and "A = {} ⟹ B = {}"
  obtains g where "g ` B = A"
proof -
  from assms obtain f where "f ` A ⊆ B" and "inj_on f A" by auto
  with assms(2) obtain g where "g ` B = A" by(elim cor_inj_on_iff_surj_on_a)
  thus "thesis" by (fact that)
qed

lemma surj_on_imp_card_leq:
  assumes "f ` A = B"
  shows "|B| ≤o |A|"
proof -
  from assms obtain g where "g ` B ⊆ A" and "inj_on g B" by (elim cor_inj_on_iff_surj_on_b)
  thus ?thesis by auto
qed

theorem thm_2_5_2_a:
  fixes Λ :: "'b set"
    and A :: "'b ⇒ 'a set"
  assumes "Λ ≠ {}"
    and "⋀l. l ∈ Λ ⟹ |A l| ≤o ℵ⇩0"
    and "|Λ| ≤o ℵ⇩0"
  shows "|⋃l ∈ Λ. A l| ≤o ℵ⇩0"
proof -
  let ?A = "⋃l ∈ Λ. A l"
  {
    assume "?A = {}"
    hence ?thesis by auto
  }
  moreover {
    assume "?A ≠ {}"
    then obtain a⇩0 where "a⇩0 ∈ ?A" by blast
    {
      fix l
      assume "l ∈ Λ"
      {
        let ?g = "λn :: nat. a⇩0"
        assume "A l = {}"
        hence "A l ⊆ ?g ` UNIV" by simp
        moreover from ‹a⇩0 ∈ ?A› have "?g ` UNIV ⊆ ?A" by auto
        ultimately have "A l ⊆ ?g ` UNIV ∧ ?g ` UNIV ⊆ ?A" ..
        hence "{g :: nat ⇒ 'a. A l ⊆ g ` UNIV ∧ g ` UNIV ⊆ ?A} ≠ {}" by fast
      }
      moreover {
        assume "A l ≠ {}"
        with ‹l ∈ Λ› assms(2) obtain g :: "nat ⇒ 'a" where "g ` UNIV = A l"
          by (metis card_leq_imp_surj_on)
        moreover from this and ‹l ∈ Λ› have "g ` UNIV ⊆ ?A" by auto
        ultimately have "A l ⊆ g ` UNIV ∧ g ` UNIV ⊆ ?A" by simp
        hence "{g :: nat ⇒ 'a. A l ⊆ g ` UNIV ∧ g ` UNIV ⊆ ?A} ≠ {}" by auto
      }
      ultimately have "{g :: nat ⇒ 'a. A l ⊆ g ` UNIV ∧ g ` UNIV ⊆ ?A} ≠ {}" by blast
    }
    then obtain f :: "'b ⇒ nat ⇒ 'a"
      where f: "f ∈ (Π l ∈ Λ. {g. A l ⊆ g ` UNIV ∧ g ` UNIV ⊆ ?A})" by (elim AC_E)
    let ?φ = "λ(l, n). f l n"
    have "?φ ` (Λ × UNIV) = ?A"
    proof (rule surj_onI; split_pair)
      fix l and  n :: nat
      assume "(l, n) ∈ Λ × UNIV"
      hence "l ∈ Λ" by simp
      with f have "f l ` UNIV ⊆ ?A" by fast
      thus "f l n ∈ ?A" by auto
    next
      fix a
      assume "a ∈ ?A"
      then obtain l where "l ∈ Λ" and "a ∈ A l" by auto
      with f obtain n where "f l n = a" by auto
      with ‹l ∈ Λ› show "a ∈ ?φ ` (Λ × UNIV)" by auto
    qed
    hence "|?A| ≤o |Λ × (UNIV :: nat set)|" by (fact surj_on_imp_card_leq)
    also from assms(3) have "|Λ × (UNIV :: nat set)| ≤o ℵ⇩0" by (auto elim: thm_2_5_1_a)
    finally have ?thesis .
  }
  ultimately show ?thesis by blast
qed

theorem thm_2_5_2_b:
  fixes Λ :: "'b set"
    and A :: "'b ⇒ 'a set"
  assumes "Λ ≠ {}"
    and "⋀l. l ∈ Λ ⟹ |A l| ≤o ℵ⇩0"
    and "|Λ| ≤o ℵ⇩0"
    and "l ∈ Λ"
    and "|A l| =o ℵ⇩0"
  shows "|⋃l ∈ Λ. A l| =o ℵ⇩0"
proof -
  let ?A = "⋃l ∈ Λ. A l"
  from assms(5) have "ℵ⇩0 =o |A l|" by (fact ordIso_symmetric)
  then obtain f :: "nat ⇒ 'a" where *: "bij_betw f UNIV (A l)" by blast
  with assms(4) have "f ` UNIV ⊆ ?A" by auto
  moreover from * have "inj_on f UNIV" by auto
  ultimately have "ℵ⇩0 ≤o |?A|" by auto
  moreover from assms(1-3) have "|?A| ≤o ℵ⇩0" by (rule thm_2_5_2_a)
  ultimately show ?thesis by (intro thm_2_3_2)
qed

lemma countable_Un_aleph_zero:
  assumes "|A| ≤o ℵ⇩0"
    and "|B| =o ℵ⇩0"
  shows "|A ∪ B| =o ℵ⇩0"
proof -
  let ?Λ = "{0 :: nat, 1}"
  let ?A = "(λl :: nat. undefined)(0 := A, 1 := B)"
  have "?Λ ≠ {}" by simp
  moreover have "⋀l. l ∈ ?Λ ⟹ |?A l| ≤o ℵ⇩0"
  proof -
    fix l
    assume "l ∈ ?Λ"
    moreover {
      assume "l = 0"
      with assms(1) have "|?A l| ≤o ℵ⇩0" by simp
    }
    moreover {
      assume "l = 1"
      from assms(2) have "|?A 1| ≤o ℵ⇩0" by fastforce
      with ‹l = 1› have "|?A l| ≤o ℵ⇩0" by simp
    }
    ultimately show "|?A l| ≤o ℵ⇩0" by blast
  qed
  moreover have "|?Λ| ≤o ℵ⇩0"
  proof -
    have "inj_on id ?Λ" by simp
    thus ?thesis by blast
  qed
  moreover have "1 ∈ ?Λ" by simp
  moreover from assms(2) have "|?A 1| =o ℵ⇩0" by simp
  ultimately have "|⋃l ∈ ?Λ. ?A l| =o ℵ⇩0" by (rule thm_2_5_2_b)
  moreover have "(⋃l ∈ ?Λ. ?A l) = A ∪ B" by simp
  ultimately show ?thesis by argo
qed

lemma aleph_zero_Un_aleph_zero:
  assumes "|A| =o ℵ⇩0"
    and "|B| =o ℵ⇩0"
  shows "|A ∪ B| =o ℵ⇩0"
proof -
  from assms(1) have "|A| ≤o ℵ⇩0" by fast
  with assms(2) show ?thesis by (intro countable_Un_aleph_zero)
qed

corollary cor_card_int_eq_aleph_zero:
  shows "|UNIV :: int set| =o ℵ⇩0"
proof -
  let ?A = "{x :: int. 0 ≤ x}"
  let ?B = "{x :: int. x ≤ 0}"
  have "|?A| =o ℵ⇩0"
  proof -
    let ?f = "λn :: nat. (of_nat n) :: int"
    have "?f ` UNIV = ?A"
    proof (rule surj_onI)
      fix n
      show "int n ∈ ?A" by simp
    next
      fix n
      assume "n ∈ ?A"
      hence "0 ≤ n" by simp
      then obtain m where "n = ?f m" by (elim nonneg_int_cases)
      thus "n ∈ ?f ` UNIV" by simp
    qed
    moreover have "inj_on ?f UNIV" by (fact inj_of_nat)
    ultimately have "bij_betw ?f UNIV ?A" by (intro bij_betw_imageI)
    hence "ℵ⇩0 =o |?A|" by blast
    thus ?thesis by (fact ordIso_symmetric)
  qed
  moreover have "|?B| =o ℵ⇩0"
  proof -
    let ?f = "λn :: nat. -((of_nat n) :: int)"
    have "?f ` UNIV = ?B"
    proof (rule surj_onI)
      fix n
      show "?f n ∈ ?B" by simp
    next
      fix n
      assume "n ∈ ?B"
      hence "0 ≤ -n" by simp
      then obtain m :: nat where "-n = m" by (elim nonneg_int_cases)
      hence "-int m = n" by simp
      thus "n ∈ ?f ` UNIV" by auto
    qed
    moreover have "inj_on ?f UNIV" by (simp add: inj_on_def)
    ultimately have "bij_betw ?f UNIV ?B" by (intro bij_betw_imageI)
    hence "ℵ⇩0 =o |?B|" by auto
    thus ?thesis by (fact ordIso_symmetric)
  qed
  ultimately have "|?A ∪ ?B| =o ℵ⇩0" by (rule aleph_zero_Un_aleph_zero)
  moreover have "?A ∪ ?B = UNIV" by auto
  ultimately show ?thesis by simp
qed

corollary cor_card_rat_eq_aleph_zero:
  shows "|UNIV :: rat set| =o ℵ⇩0"
proof -
  let ?f = "λ(a, b). Fract a b"
  have "?f ` (UNIV × UNIV) = UNIV"
  proof (rule surj_onI; split_pair)
    fix a b
    show "Fract a b ∈ UNIV" by simp
  next
    fix q
    obtain a b where "q = Fract a b" by (auto intro: Rat_cases)
    thus "q ∈ ?f ` (UNIV × UNIV)" by auto
  qed
  hence "|UNIV :: rat set| ≤o |(UNIV :: int set) × (UNIV :: int set)|"
    by (fact surj_on_imp_card_leq)
  also have "|(UNIV :: int set) × (UNIV :: int set)| =o ℵ⇩0"
  proof -
    have "|UNIV :: int set| =o ℵ⇩0" by (fact cor_card_int_eq_aleph_zero)
    moreover from this have "|UNIV :: int set| ≤o ℵ⇩0" by fast
    ultimately show "|(UNIV :: int set) × (UNIV :: int set)| =o ℵ⇩0" by (blast intro: thm_2_5_1_b)
  qed
  finally have "|UNIV :: rat set| ≤o ℵ⇩0" .
  moreover have "ℵ⇩0 ≤o |UNIV :: rat set|"
  proof -
    let ?g = "λn :: nat. (of_nat n) :: rat"
    have "inj_on ?g UNIV" by (auto intro: inj_of_nat)
    thus ?thesis by blast
  qed
  ultimately show ?thesis by (intro thm_2_3_2)
qed

theorem thm_2_6:
  assumes "infinite A"
    and "B ⊆ A"
    and "|B| ≤o ℵ⇩0"
    and "infinite (A - B)"
  shows "equipotent (A - B) A"
proof -
  let ?A⇩1 = "A - B"
  from assms(4) obtain C where "C ⊆ ?A⇩1" and "|C| =o ℵ⇩0" by (elim thm_2_4)
  let ?A⇩2 = "?A⇩1 - C"
  from assms(3) and ‹|C| =o ℵ⇩0› have "|B ∪ C| =o ℵ⇩0" by (rule countable_Un_aleph_zero)
  also from ‹|C| =o ℵ⇩0› have "ℵ⇩0 =o |C|" by (fact ordIso_symmetric)
  finally have "|B ∪ C| =o |C|" .
  then obtain f⇩1 where "bij_betw f⇩1 (B ∪ C) C" by auto
  let ?f = "λa. if a ∈ ?A⇩2 then a else f⇩1 a"
  have "?f ` A = ?A⇩1"
  proof (rule surj_onI)
    fix a
    assume "a ∈ A"
    {
      assume "a ∈ ?A⇩2"
      hence "?f a ∈ ?A⇩1" by simp
    }
    moreover {
      assume "a ∉ ?A⇩2"
      with ‹a ∈ A› have "a ∈ B ∪ C" by simp
      from ‹a ∉ ?A⇩2› have "?f a = f⇩1 a"by argo
      also from ‹a ∈ B ∪ C› and ‹bij_betw f⇩1 (B ∪ C) C› have "… ∈ C" by auto
      also from ‹C ⊆ ?A⇩1› have "… ⊆ ?A⇩1" by simp
      finally have "?f a ∈ ?A⇩1" .
    }
    ultimately show "?f a ∈ ?A⇩1" by blast
  next
    fix b
    assume "b ∈ ?A⇩1"
    {
      assume "b ∈ C"
      with ‹bij_betw f⇩1 (B ∪ C) C› obtain a where "a ∈ B ∪ C" and "b = f⇩1 a" by auto
      from this(1) have "a ∉ ?A⇩2" by simp
      with ‹b = f⇩1 a› have "?f a = b" by argo
      moreover from ‹a ∈ B ∪ C› and assms(2) and ‹C ⊆ ?A⇩1› have "a ∈ A" by auto
      ultimately have "∃a ∈ A. ?f a = b" by blast
    }
    moreover {
      assume "b ∉ C"
      with ‹b ∈ ?A⇩1› have "b ∈ ?A⇩2" by simp
      hence "?f b = b" by simp
      moreover from ‹b ∈ ?A⇩1› have "b ∈ A" by simp
      ultimately have "∃a ∈ A. ?f a = b" by blast
    }
    ultimately show "b ∈ ?f ` A" by blast
  qed
  moreover have "inj_on ?f A"
  proof (rule inj_onI)
    fix a a'
    assume "a ∈ A" and "a' ∈ A" and "?f a = ?f a'"
    consider (A) "a ∈ ?A⇩2" and "a' ∈ ?A⇩2"
      | (B) "a ∈ ?A⇩2" and "a' ∉ ?A⇩2"
      | (C) "a ∉ ?A⇩2" and "a' ∈ ?A⇩2"
      | (D) "a ∉ ?A⇩2" and "a' ∉ ?A⇩2" by argo
    thus "a = a'"
    proof cases
      case A
      with ‹?f a = ?f a'› show ?thesis by simp
    next
      case B
      from B(2) and ‹a' ∈ A› have "a' ∈ B ∪ C" by simp
      with ‹?f a = ?f a'› B have "a = f⇩1 a'" by meson
      with ‹bij_betw f⇩1 (B ∪ C) C› and ‹a' ∈ B ∪ C› and B(1) have "False" by fast
      thus ?thesis ..
    next
      case C
      from C(1) and ‹a ∈ A› have "a ∈ B ∪ C" by simp
      with ‹?f a = ?f a'› C have "f⇩1 a = a'" by metis
      with ‹bij_betw f⇩1 (B ∪ C) C› and ‹a ∈ B ∪ C› and C(2) have "False" by auto
      thus ?thesis ..
    next
      case D
      with ‹?f a = ?f a'› have "f⇩1 a = f⇩1 a'" by argo
      moreover from D and ‹a ∈ A› and ‹a' ∈ A› have "a ∈ B ∪ C" and "a' ∈ B ∪ C" by blast+
      moreover from ‹bij_betw f⇩1 (B ∪ C) C› have "inj_on f⇩1 (B ∪ C)" by auto
      ultimately show ?thesis by (elim inj_onD)
    qed
  qed
  ultimately have "bij_betw ?f A ?A⇩1" by (intro bij_betw_imageI)
  hence "equipotent A ?A⇩1" by auto
  thus ?thesis by (fact prop_2_1_2)
qed

corollary cor_2_1:
  assumes "infinite A"
    and "|B| ≤o ℵ⇩0"
  shows "equipotent (A ∪ B) A"
proof -

  qed

qed

theorem thm_2_8:
  shows "|M| <o |Pow M|"

    }

      assume "x ∉ ?X"
      with ‹x ∈ M› have "x ∈ f x" by simp
      with ‹f x = ?X› have "x ∈ ?X" by blast
      with ‹x ∉ ?X› have "False" by simp
    }
    ultimately show "False" by auto
  qed

proposition prob_2_2_2:
  fixes A :: "'a set"
  assumes "|A| =o ℵ⇩0"
  obtains A' :: "nat ⇒ 'a set" where "⋀n. |A' n| =o ℵ⇩0"
    and "(⋃n. A' n) = A"
    and "disjoint_family_on A' UNIV"
proof -
  have "|(UNIV :: nat set) × (UNIV :: nat set)| =o ℵ⇩0" by (fact nat_Times_nat_card_eq_aleph_zero)
  also from assms have "ℵ⇩0 =o |A|" by (fact ordIso_symmetric)
  finally have "|(UNIV :: nat set) × (UNIV :: nat set)| =o |A|" .
  then obtain f :: "nat × nat ⇒ 'a" where f: "bij_betw f (UNIV × UNIV) A" by auto
  hence inj_f: "inj f" by auto
  let ?A' = "λn :: nat. f ` ({n} × UNIV)"
  have "⋀n. |?A' n| =o ℵ⇩0"
  proof -
    fix m
    let ?g = "λa. THE n. f (m, n) = a"
    have g: "?g a = n" if "f (m, n) = a" for n and a
    proof -
      {
        fix n'
        assume "f (m, n') = a"
        with that have "f (m, n') = f (m, n)" by simp
        with inj_f have "n' = n" by (auto dest: injD)
      }
      with that show ?thesis by blast
    qed
    have "?g ` (?A' m) = UNIV"
    proof (rule surj_onI)
      fix a
      assume "a ∈ ?A' m"
      show "?g a ∈ UNIV" by simp
    next
      fix n
      have "f (m, n) ∈ ?A' m" by simp
      moreover have "?g (f (m, n)) = n"
      proof (rule the_equality, rule refl)
        fix n'
        assume "f (m, n') = f (m, n)"
        with inj_f have "(m, n') = (m, n)" by (auto dest: injD)
        thus "n' = n" by simp
      qed
      ultimately show "n ∈ ?g ` ?A' m" by fastforce
    qed
    moreover have "inj_on ?g (?A' m)"
    proof (rule inj_onI)
      fix a and a'
      assume "a ∈ ?A' m" and "a' ∈ ?A' m" and "?g a = ?g a'"
      from this(1,2) obtain n and n' where "f (m, n) = a" and "f (m, n') = a'" by auto
      moreover from this and ‹?g a = ?g a'› have "n = n'" using g by simp+
      ultimately show "a = a'" by simp
    qed
    ultimately show "|?A' m| =o ℵ⇩0" by (blast intro: bij_betw_imageI)
  qed
  moreover have "(⋃n. ?A' n) = A"
  proof (rule set_eqI2)
    fix a
    assume "a ∈ (⋃n. ?A' n)"
    then obtain m where "a ∈ ?A' m" by auto
    then obtain n where "a = f (m, n)" by auto
    also from f have "f (m, n) ∈ A" by blast
    finally show "a ∈ A" .
  next
    fix a
    assume "a ∈ A"
    with f obtain m and n where "a = f (m, n)" by blast
    hence "a ∈ ?A' m" by simp
    thus "a ∈ (⋃n. ?A' n)" by blast
  qed
  moreover have "disjoint_family_on ?A' UNIV"
  proof (rule disjoint_family_onI)
    fix m m' :: "nat"
    assume "m ≠ m'"
    {
      fix a
      assume "a ∈ ?A' m" and "a ∈ ?A' m'"
      then obtain n n' :: "nat" where "a = f (m, n)" and "a = f (m', n')" by blast
      hence "f (m, n) = f (m', n')" by simp
      with inj_f have "(m, n) = (m', n')" by (auto dest: injD)
      with ‹m ≠ m'› have "False" by simp
    }
    thus "?A' m ∩ ?A' m' = {}" by fast
  qed
  ultimately show "thesis" by (fact that)
qed

lemma fixed_length_lists_of_aleph_zero:
  fixes A :: "'a set"
  assumes "|A| =o ℵ⇩0"
    and "1 ≤ n"
  defines XS: "XS l ≡ {xs ∈ lists A. length xs = l}"
  shows "|XS n| =o ℵ⇩0"
using assms(2) proof (induct n rule: dec_induct)
  case base
  let ?f = "λa. [a]"
  have "?f ` A = XS 1"
  proof (rule surj_onI)
    fix a
    assume "a ∈ A"
    thus "?f a ∈ XS 1" unfolding XS by simp
  next
    fix xs
    assume "xs ∈ XS 1"
    hence "xs ∈ lists A" and "length xs = 1" unfolding XS by simp+
    from this(2) have "length xs = Suc 0" by simp
    then obtain y and ys where "xs = y # ys" and "length ys = 0" by (auto simp: length_Suc_conv)
    from this(2) have "ys = []" by simp
    with ‹xs = y # ys› have "xs = [y]" by simp
    moreover from ‹xs = y # ys› and ‹xs ∈ lists A› have "y ∈ A" by simp
    ultimately show "xs ∈ ?f ` A" by simp
  qed
  moreover have "inj_on ?f A"
  proof (rule inj_onI)
    fix a a' :: "'a"
    assume "?f a = ?f a'"
    thus "a = a'" by simp
  qed
  ultimately have "bij_betw ?f A (XS 1)" by (intro bij_betw_imageI)
  hence "|A| =o |XS 1|" by auto
  hence "|XS 1| =o |A|" by (fact ordIso_symmetric)
  also note assms(1)
  finally show "?case" .
next
  case (step n)
  let ?f = "λ(xs, x). x # xs"
  have "?f ` ((XS n) × A) = XS (Suc n)"
  proof (rule surj_onI; split_pair)
    fix xs and a
    assume "(xs, a) ∈ (XS n) × A"
    hence "xs ∈ XS n" and "a ∈ A" by simp+
    from this(1) have "xs ∈ lists A" and "length xs = n" unfolding XS by simp+
    from this(1) and ‹a ∈ A› have "a # xs ∈ lists A" by simp
    moreover from ‹length xs = n› have "length (a # xs) = Suc n" by simp
    ultimately show "a # xs ∈ XS (Suc n)" unfolding XS by simp
  next
    fix xs
    assume "xs ∈ XS (Suc n)"
    hence "xs ∈ lists A" and "length xs = Suc n" unfolding XS by simp+
    from this(2) obtain a and ys where "xs = a # ys" and "length ys = n"
      by (auto simp: length_Suc_conv)
    from ‹xs ∈ lists A› and this(1) have "ys ∈ lists A" by simp
    with ‹length ys = n› have "ys ∈ XS n" unfolding XS by simp
    moreover from ‹xs ∈ lists A› and ‹xs = a # ys› have "a ∈ A" by simp
    moreover note ‹xs = a # ys›
    ultimately show "xs ∈ ?f ` (XS n × A)" by auto
  qed
  hence "|XS (Suc n)| ≤o |(XS n) × A|" by (fact surj_on_imp_card_leq)
  also from assms(1) and step.hyps have "|(XS n) × A| =o ℵ⇩0"
    by (intro aleph_zero_Times_aleph_zero)
  finally have "|XS (Suc n)| ≤o ℵ⇩0" .
  moreover have "ℵ⇩0 ≤o |XS (Suc n)|"
  proof -
    from step.hyps(3) have "ℵ⇩0 =o |XS n|" by (fact ordIso_symmetric)
    also have "|XS n| ≤o |XS (Suc n)|"
    proof -
      from assms(1) obtain a where "a ∈ A" by blast
      let ?f = "λxs. a # xs"
      have "?f ` (XS n) ⊆ XS (Suc n)"
      proof (rule image_subsetI)
        fix xs
        assume "xs ∈ XS n"
        hence "xs ∈ lists A" and "length xs = n" unfolding XS by simp+
        from this(1) and ‹a ∈ A› have "?f xs ∈ lists A" by simp
        moreover from ‹length xs = n› have "length (?f xs) = Suc n" by simp
        ultimately show "?f xs ∈ XS (Suc n)" unfolding XS by simp
      qed
      moreover have "inj_on ?f (XS n)" by simp
      ultimately show ?thesis by blast
    qed
    finally show ?thesis .
  qed
  ultimately show "?case" by (fact thm_2_3_2)
qed

lemma lists_aleph_zero_eq_aleph_zero:
  assumes "|A| =o ℵ⇩0"
  shows "|lists A| =o ℵ⇩0"
proof -
  let ?A' = "λn. {xs ∈ lists A. length xs = n}"
  let ?B = "⋃i. ?A' i"
  have "lists A ⊆ ?B"
  proof (rule subsetI)
    fix xs
    assume "xs ∈ lists A"
    hence "xs ∈ ?A' (length xs)" by simp
    thus "xs ∈ ?B" by blast
  qed
  hence "|lists A| ≤o |?B|" by fastforce
  also have "|?B| =o ℵ⇩0"
  proof (rule thm_2_5_2_b; simp?)
    fix n :: nat
    {
      assume "n = 0"
      hence "?A' n = {[]}" by auto
      hence "|?A' n| ≤o ℵ⇩0" by auto
    }
    moreover {
      assume "1 ≤ n"
      with assms have "|?A' n| =o ℵ⇩0" by (fact fixed_length_lists_of_aleph_zero)
      hence "|?A' n| ≤o ℵ⇩0" by (fact card_eq_imp_card_leq)
    }
    ultimately show "|?A' n| ≤o ℵ⇩0" by linarith
  next
    from assms show "|?A' 1| =o ℵ⇩0" by (blast intro: fixed_length_lists_of_aleph_zero)
  qed
  finally have "|lists A| ≤o ℵ⇩0" by simp
  moreover have "ℵ⇩0 ≤o |lists A|"
  proof -
    from assms have "|?A' 1| =o ℵ⇩0" by (blast intro: fixed_length_lists_of_aleph_zero)
    hence "ℵ⇩0 =o |?A' 1|" by (fact ordIso_symmetric)
    also have "|?A' 1| ≤o |lists A|"
    proof -
      have "?A' 1 ⊆ lists A" by auto
      thus ?thesis by auto
    qed
    finally show ?thesis .
  qed
  ultimately show ?thesis by (fact thm_2_3_2)
qed

proposition prob_2_2_5:
  assumes "|A| =o ℵ⇩0"
  defines AA: "𝔄 ≡ {X. X ⊆ A ∧ finite X}"
  shows "|𝔄| =o ℵ⇩0"
proof -
  let ?A' = "λn. {xs ∈ lists A. length xs = n}"
  let ?B = "⋃n. ?A' n"

    fix xs
    assume "xs ∈ ?B"
    then obtain n where "xs ∈ ?A' n" by simp
    hence "xs ∈ lists A" by simp
    hence "?f xs ⊆ A" by auto
    moreover have "finite (?f xs)" by simp
    ultimately show "?f xs ∈ 𝔄" unfolding AA by simp
  next
    fix X
    assume "X ∈ 𝔄"
    hence "X ⊆ A" and "finite X" unfolding AA by simp+
    then obtain xs where "xs ∈ lists A" and "set xs = X" by (fast dest: finite_list)
    from this(1) have "xs ∈ ?A' (length xs)" by simp
    hence "xs ∈ ?B" by simp
    with ‹set xs = X› show "X ∈ ?f ` ?B" by auto
  qed
  hence "|𝔄| ≤o |?B|" by (fact surj_on_imp_card_leq)
  also have "|?B| ≤o ℵ⇩0"
  proof (rule thm_2_5_2_a; simp)
    fix n
    have "?A' n ⊆ lists A" by auto
    hence "|?A' n| ≤o |lists A|" by auto
    also from assms(1) have "|lists A| =o ℵ⇩0" by (fact lists_aleph_zero_eq_aleph_zero)
    finally show "|?A' n| ≤o ℵ⇩0" .
  qed
  finally have "|𝔄| ≤o ℵ⇩0" .
  moreover have "ℵ⇩0 ≤o |𝔄|"
  proof -
    from assms(1) have "ℵ⇩0 =o |A|" by (fact ordIso_symmetric)
    also have "|A| ≤o |𝔄|"
    proof (rule card_leqI)
      let ?f = "λa :: 'a. {a}"
      {
        fix a
        assume "a ∈ A"
        hence "?f a ∈ 𝔄" unfolding AA by simp
      }
      thus "?f ` A ⊆ 𝔄" by auto
      show "inj_on ?f A" by simp
    qed
    finally show ?thesis .
  qed
  ultimately show ?thesis by (fact thm_2_3_2)
qed
