Theory Split_Pair

theory Split_Pair
imports Main Eisbach
theory Split_Pair
  imports Main
    "HOL-Eisbach.Eisbach"
begin

lemma split_paired_Coll:
  shows "{x. P x} = {(a, b). P (a, b)}"
  by simp

lemma split_paired_Collect:
  shows "{x ∈ X. P x} = {(a, b) ∈ X. P (a, b)}"
  by auto

lemma split_paired_Ball_Times:
  shows "(∀x ∈ A × B. P x) ⟷ (∀a ∈ A. ∀b ∈ B. P (a, b))"
  by simp

lemma split_paired_Bex_Times:
  shows "(∃x ∈ A × B. P x) ⟷ (∃a ∈ A. ∃b ∈ B. P (a, b))"
  by simp

method split_pair = (
    (unfold split_paired_all)?;
    (unfold split_paired_All)?;
    (unfold split_paired_Ex)?;
    (unfold split_paired_The)?;
    (subst split_paired_Coll)?;
    (subst split_paired_Collect)?;
    (unfold split_paired_Ball_Times)?;
    (unfold split_paired_Bex_Times)?;
    (unfold case_prod_conv)?)

experiment
  fixes A :: "('a × 'b) set"
begin

lemma
  shows "A = A"
proof (rule set_eqI; split_pair)
  fix a b
  show "(a, b) ∈ A ⟷ (a, b) ∈ A" ..
qed

lemma subsetI_atomize:
  assumes "∀x. x ∈ X ⟶ x ∈ Y"
  shows "X ⊆ Y"
  using assms by auto

lemma
  assumes "A = {}"
    and "B = {}"
  shows "A ⊆ B"
proof (rule subsetI_atomize; split_pair)
  {
    fix a b
    assume "(a, b) ∈ A"
    with assms(1) have "(a, b) ∈ B" by simp
  }
  thus "∀a b. (a, b) ∈ A ⟶ (a, b) ∈ B" by simp
qed

lemma ex_mem_imp_not_empty:
  assumes "∃x. x ∈ X"
  shows "X ≠ {}"
  using assms by auto

lemma
  assumes "(a, b) ∈ A"
  shows "A ≠ {}"
proof (rule ex_mem_imp_not_empty; split_pair)
  from assms show "∃a b. (a, b) ∈ A" by auto
qed

lemma Coll_False_imp_empty:
  assumes "X = {x. False}"
  shows "X = {}"
  using assms by simp

lemma
  assumes "A = {(a, b). a ≠ a}"
  shows "A = {}"
proof (rule Coll_False_imp_empty; split_pair)
  from assms show "A = {(a, b). False}" by simp
qed

lemma Collect_False_imp_empty:
  assumes "X = {x ∈ Y. False}"
  shows "X = {}"
  using assms by simp

lemma
  assumes "A = {(a, b) ∈ Y. a ≠ a}"
  shows "A = {}"
proof (rule Collect_False_imp_empty; split_pair)
  from assms show "A = {(a, b) ∈ Y. False}" by simp
qed

lemma subsetI_Ball:
  assumes "∀x ∈ X. x ∈ Y"
  shows "X ⊆ Y"
  using assms by auto

lemma
  assumes "⋀a b. (a, b) ∈ X × Y ⟹ (a, b) ∈ Z × W"
  shows "(X × Y) ⊆ (Z × W)"
proof (rule subsetI_Ball; split_pair)
  {
    fix a b
    assume "a ∈ X"
      and "b ∈ Y"
    with assms have "(a, b) ∈ Z × W" by simp
  }
  thus "∀a ∈ X. ∀ b ∈ Y. (a, b) ∈ Z × W" by simp
qed

lemma Bex_imp_not_empty:
  assumes "∃x ∈ X. P x"
  shows "X ≠ {}"
  using assms by auto

lemma
  assumes "x ∈ X × Y"
  shows "X × Y ≠ {}"
proof (rule Bex_imp_not_empty[where P = "λx. True"]; split_pair)
  from assms have "fst x ∈ X" and "snd x ∈ Y" by auto
  thus "∃x ∈ X. ∃y ∈ Y. True" by auto
qed

end

end
