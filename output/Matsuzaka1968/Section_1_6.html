lemma refl_onE [elim]:
  assumes "refl_on A r"
  obtains "a ∈ A ⟹ (a, a) ∈ r"
    and "(x, y) ∈ r ⟹ x ∈ A"
    and "(x, y) ∈ r ⟹ y ∈ A"
  using assms by (blast dest: refl_onD refl_on_domain)

definition equiv_kernel_on :: "('a ⇒ 'b) ⇒ 'a set ⇒ ('a × 'a) set"
  where "equiv_kernel_on f A = {(a, a') ∈ A × A. f a = f a'}"

lemma equiv_kernel_onI [intro]:
  assumes "a ∈ A"
    and "a' ∈ A"
    and "f a = f a'"
  shows "(a, a') ∈ equiv_kernel_on f A"
  using assms unfolding equiv_kernel_on_def by simp

lemma equiv_kernel_onE [elim]:
  assumes "(a, a') ∈ equiv_kernel_on f A"
  obtains "a ∈ A"
    and "a' ∈ A"
    and "f a = f a'"
  using assms unfolding equiv_kernel_on_def by simp

lemma equiv_equiv_kernel_on:
  shows "equiv A (equiv_kernel_on f A)"
proof (rule equivI)
  show "refl_on A (equiv_kernel_on f A)"
  proof (rule refl_onI)
    show "equiv_kernel_on f A ⊆ A × A" by auto
    fix a
    assume "a ∈ A"
    moreover have "f a = f a" by simp
    ultimately show "(a, a) ∈ equiv_kernel_on f A" by auto
  qed
  show "sym (equiv_kernel_on f A)"
  proof (rule symI)
    fix a and a'
    assume "(a, a') ∈ equiv_kernel_on f A"
    thus "(a', a) ∈ equiv_kernel_on f A" by fastforce
  qed
  show "trans (equiv_kernel_on f A)"
  proof (rule transI)
    fix a and a' and a''
    assume "(a, a') ∈ equiv_kernel_on f A" and "(a', a'') ∈ equiv_kernel_on f A"
    thus "(a, a'') ∈ equiv_kernel_on f A" by fastforce
  qed
qed

    and "⋀M M'. M ∈ 𝔐 ⟹ M' ∈ 𝔐 ⟹ M ≠ M' ⟹ M ∩ M' = {}"

lemma partition_onE [elim]:
  assumes "partition_on A 𝔐"
  obtains "{} ∉ 𝔐"
    and "⋃𝔐 = A"
    and "M ∈ 𝔐 ⟹ M' ∈ 𝔐 ⟹ M ≠ M' ⟹ M ∩ M' = {}"

lemma partition_on_definition:
  shows "partition_on A 𝔐
           ⟷ (∀M ∈ 𝔐. M ≠ {}) ∧ ⋃𝔐 = A ∧ (∀M ∈ 𝔐. ∀M' ∈ 𝔐. M ≠ M' ⟶ M ∩ M' = {})"
    (is "?L ⟷ ?R0 ∧ ?R1 ∧ ?R2")
proof (rule iffI)
  assume "?L"
  thus "?R0 ∧ ?R1 ∧ ?R2" by fast
next
  assume *: "?R0 ∧ ?R1 ∧ ?R2"
  from * have "?R1" by simp
  moreover {
    fix M and M'
    assume "M ∈ 𝔐" and "M' ∈ 𝔐" and "M ≠ M'"
    with * have "disjnt M M'" unfolding disjnt_def by simp
  }
  moreover from * have "{} ∉ 𝔐" by auto
  ultimately show "partition_on A 𝔐" by (fact partition_onI)
qed

proposition ex_1_1_a:
  shows "equiv A {(a, a') ∈ A × A. a = a'}"
proof (rule equivI)
  let ?R = "{(a, a') ∈ A × A. a = a'}"
  show "refl_on A ?R"
  proof (rule refl_onI)
    show "?R ⊆ A × A" by auto
    fix a
    assume "a ∈ A"
    hence "(a, a) ∈ A × A" by simp
    moreover have "a = a" by simp
    ultimately show "(a, a) ∈ ?R" by simp
  qed
  show "sym ?R"
  proof (rule symI)
    fix a and a'
    assume "(a, a') ∈ {(a, a') ∈ A × A. a = a'}"
    hence "(a', a) ∈ A × A" and "a' = a" by simp+
    thus "(a', a) ∈ {(a, a') ∈ A × A. a = a'}" by simp
  qed
  show "trans ?R"
  proof (rule transI)
    fix a and a' and a''
    assume "(a, a') ∈ ?R" and "(a', a'') ∈ ?R"
    hence "(a, a'') ∈ A × A" and "a = a''" by simp+
    thus "(a, a'') ∈ ?R" by simp
  qed
qed

proposition prop_1_6_4:
  assumes "f ` A ⊆ B"
  obtains φ and g and j where
    "φ ` A = A // (equiv_kernel_on f A)"
    and "bij_betw g (A // (equiv_kernel_on f A)) (f ` A)"
    and "id_on j (f ` A)"
    and "ext_eq_on A (j ∘ g ∘ φ) f"
proof -
  have *: "equiv A (equiv_kernel_on f A)" by (fact equiv_equiv_kernel_on)
  let ?φ = "λa. equiv_kernel_on f A `` {a}"
  let ?g = "λC. (THE b. ∃a ∈ A. ?φ a = C ∧ f a = b)"
  let ?j = "λb. if b ∈ f ` A then b else undefined"
  have **: "?g (?φ a) = f a" if "a ∈ A" for a
  proof (rule the_equality)
    from that show "∃a' ∈ A. ?φ a' = equiv_kernel_on f A `` {a} ∧ f a' = f a" by blast
    fix b
    assume "∃a' ∈ A. ?φ a' = equiv_kernel_on f A `` {a} ∧ f a' = b"
    then obtain a' where "a' ∈ A" and "?φ a' = equiv_kernel_on f A `` {a}" and "f a' = b" by auto
    from * and this(1,2) have "(a', a) ∈ equiv_kernel_on f A" by (rule prop_1_6_2_b)
    hence "f a' = f a" by auto
    with ‹f a' = b› show "b = f a" by simp
  qed
  have "?φ ` A = A // (equiv_kernel_on f A)"
  proof (rule surj_onI)
    fix a
    assume "a ∈ A"
    thus "?φ a ∈ A // (equiv_kernel_on f A)" by auto
  next
    fix C
    assume "C ∈ A // (equiv_kernel_on f A)"
    then obtain a where "a ∈ A" and "C = equiv_kernel_on f A `` {a}" by fast
    thus "C ∈ ?φ ` A" by simp
  qed
  moreover have "bij_betw ?g (A // (equiv_kernel_on f A)) (f ` A)"
  proof (rule bij_betw_imageI)
    show "?g ` (A // (equiv_kernel_on f A)) = f ` A"
    proof (rule surj_onI)
      fix C
      assume "C ∈ A // (equiv_kernel_on f A)"
      then obtain a where "a ∈ A" and "C = ?φ a" by fast
      from this(1) have "?g (?φ a) = f a" by (fact **)
      with ‹C = ?φ a› have "?g C = f a" by simp
      with ‹a ∈ A› show "?g C ∈ f ` A" by simp
    next
      fix b
      assume "b ∈ f ` A"
      then obtain a where "a ∈ A" and "b = f a" by auto
      from this(1) have "?g (?φ a) = f a" by (fact **)
      with ‹b = f a› have "?g (?φ a) = b" by simp
      moreover from ‹a ∈ A› have "?φ a ∈ A // equiv_kernel_on f A" by auto
      ultimately show "b ∈ ?g ` (A // (equiv_kernel_on f A))" by auto
    qed
    show "inj_on ?g (A // (equiv_kernel_on f A))"
    proof (rule inj_onI)
      fix C and C'
      assume "C ∈ A // (equiv_kernel_on f A)"
        and "C' ∈ A // (equiv_kernel_on f A)"
        and "?g C = ?g C'"
      from this(1,2) obtain a and a' where "a ∈ A" and "C = ?φ a" "a' ∈ A" and "C' = ?φ a'"
        by fastforce
      from this have "?g (?φ a) = f a" and "?g (?φ a') = f a'" by (auto intro: **)
      with ‹?g C = ?g C'› and ‹C = ?φ a› and ‹C' = ?φ a'› have "f a = f a'" by force
      with ‹a ∈ A› and ‹a' ∈ A› have "?φ a = ?φ a'" by fastforce
      with ‹C = ?φ a› and ‹C' = ?φ a'› show "C = C'" by simp
    qed
  qed
  moreover have "id_on ?j (f ` A)"
  proof (rule id_onI)
    fix b
    assume "b ∈ f ` A"
    thus "?j b = b" by simp
  qed
  moreover have "ext_eq_on A (?j ∘ ?g ∘ ?φ) f"
  proof (rule ext_eq_onI)
    fix a
    assume "a ∈ A"
    hence "?g (?φ a) = f a" by (fact **)
    with ‹a ∈ A› have "?g (?φ a) ∈ f ` A" by simp
    hence "?j (?g (?φ a)) = ?g (?φ a)" by simp
    with ‹?g (?φ a) = f a› show "(?j ∘ ?g ∘ ?φ) a = f a" by force
  qed
  ultimately show "thesis" by (fact that)
qed

proposition prob_1_6_1_a:
  assumes "A - B ≠ {}"
    and "B - A ≠ {}"
    and "A ∩ B ≠ {}"
  defines R: "R ≡ A × A ∪ B × B"
  shows "refl_on (A ∪ B) R"
    and "sym R"
    and "¬trans R"
proof -
  have "R ⊆ (A ∪ B) × (A ∪ B)" unfolding R by auto
  moreover {
    fix a
    assume "a ∈ A ∪ B"
    hence "(a, a) ∈ R" unfolding R by simp
  }
  ultimately show "refl_on (A ∪ B) R" by (rule refl_onI)
  {
    fix a and a'
    assume "(a, a') ∈ R"
    hence "(a', a) ∈ R" unfolding R by auto
  }
  thus "sym R" by (rule symI)
  {
    assume "trans R"
    from assms obtain a and b and c where "a ∈ A - B" and "b ∈ B - A" and "c ∈ A ∩ B" by auto
    hence "(a, c) ∈ R" and "(c, b) ∈ R" and "(a, b) ∉ R" unfolding R by auto
    from this(1,2) and ‹trans R› have "(a, b) ∈ R" by auto
    with ‹(a, b) ∉ R› have "False" ..
  }
  thus "¬trans R" ..
qed

proposition prob_1_6_1_b:
  defines R: "R :: nat rel ≡ {(a, b). a ≤ b}"
  shows "refl_on UNIV R"
    and "¬sym R"
    and "trans R"
proof -
  have "R ⊆ UNIV × UNIV" unfolding R by simp
  moreover {
    fix a
    have "(a, a) ∈ R" unfolding R by simp
  }
  ultimately show "refl_on UNIV R" by (rule refl_onI)
  {
    assume "sym R"
    moreover have "(0, 1) ∈ R" unfolding R by simp
    ultimately have "(1, 0) ∈ R" by auto
    moreover have "(1, 0) ∉ R" unfolding R by simp
    ultimately have "False" by simp
  }
  thus "¬sym R" by auto
  {
    fix a and a' and a''
    assume "(a, a') ∈ R" and "(a', a'') ∈ R"
    hence "(a, a'') ∈ R" unfolding R by fastforce
  }
  thus "trans R" by (rule transI)
qed

proposition prob_1_6_2:

    and "sym R"
    and "trans R"

  }

qed

proposition prob_1_6_3:
  assumes ― ‹The assumption @{prop "R ⊆ A × A"} is not necessary since it is implied by
              @{prop "refl_on A R"}.›
    "refl_on A R"
    and "∀a b c. (a, b) ∈ R ∧ (b, c) ∈ R ⟶ (c, a) ∈ R"
  shows "equiv A R"
proof (rule equivI[OF assms(1)])
  {
    fix a and a'
    assume "(a, a') ∈ R"
    moreover from this and assms(1) have "a ∈ A" and "a' ∈ A" by auto
    moreover from this(2) and assms(1) have "(a', a') ∈ R" by auto
    moreover note assms(2)
    ultimately have "(a', a) ∈ R" by blast
  }
  thus "sym R" by (rule symI)
  {
    fix a and a' and a''
    assume "(a, a') ∈ R" and "(a', a'') ∈ R"
    with assms(2) have "(a'', a) ∈ R" by blast
    with ‹sym R› have "(a, a'') ∈ R" by auto
  }
  thus "trans R" by (rule transI)
qed

proposition prob_1_6_4:
  defines A: "A :: int rel ≡ UNIV × (UNIV - {0})"
  defines R: "R ≡ {((m, n), (m', n')) ∈ A × A. m * n' = m' * n}"
  shows "equiv A R"
proof (rule equivI)
  have "R ⊆ A × A" unfolding R by auto
  moreover {
    fix m and n
    assume "(m, n) ∈ A"
    hence "((m, n), (m, n)) ∈ R" unfolding R by simp
  }
  ultimately show "refl_on A R" by (intro refl_onI; split_pair)
  {
    fix m and n and m' and n'
    assume "((m, n), (m', n')) ∈ R"
    hence "(m, n) ∈ A" and "(m', n') ∈ A" and "m * n' = m' * n" unfolding R by auto
    from this(3) have "m' * n = m * n'" by simp
    with ‹(m, n) ∈ A› and ‹(m', n') ∈ A› have "((m', n'), (m, n)) ∈ R" unfolding R by fast
  }
  thus "sym R" by (intro symI, split_pair)
  {
    fix m and n and m' and n' and m'' and n''
    assume "((m, n), (m', n')) ∈ R" and "((m', n'), (m'', n'')) ∈ R"
    hence "(m, n) ∈ A"
      and "(m', n') ∈ A"
      and "(m'', n'') ∈ A"
      and "m * n' = m' * n"
      and "m' * n'' = m'' * n'" unfolding R by auto
    from this(4,5) have "m * n' * n'' = m' * n * n''"
      and "m' * n * n'' = m'' * n * n'" by simp+
    hence "m * n'' * n' = m'' * n * n'" by algebra
    moreover from ‹(m', n') ∈ A› have "n' ≠ 0" unfolding A by simp
    ultimately have "m * n'' = m'' * n" by simp
    with ‹(m, n) ∈ A› and ‹(m'', n'') ∈ A› have "((m, n), (m'', n'')) ∈ R" unfolding R by simp
  }
  thus "trans R" by (intro transI, split_pair)
qed

proposition prob_1_6_6_a:
  assumes "equiv A R"
  defines phi: "φ a ≡ R `` {a}"
  assumes ― ‹The assumption @{prop "f ` A ⊆ B"} is not necessary.›
    ― ‹The assumption @{prop "g ` (A // R) ⊆ B"} is not necessary.›
    "ext_eq_on A f (g ∘ φ)"
  shows "∀a ∈ A. ∀a' ∈ A. (a, a') ∈ R ⟶ f a = f a'"
proof (intro ballI, rule impI)
  fix a and a'
  assume "(a, a') ∈ R"
  with assms(1) have "R `` {a} = R `` {a'}" by (rule prop_1_6_2_a)
  hence "φ a = φ a'" by (unfold phi)
  moreover from assms(1) and ‹(a, a') ∈ R› have "a ∈ A" and "a' ∈ A" by auto
  moreover note assms(3)
  ultimately show "f a = f a'" by (elim prob_1_5_14_a)
qed

proposition prob_1_6_6_b:
  assumes "equiv A R"
  defines phi: "φ ≡ λa. R `` {a}"
  assumes "f ` A ⊆ B"
    and "∀a ∈ A. ∀a' ∈ A. (a, a') ∈ R ⟶ f a = f a'"
  obtains g where "g ` (A // R) ⊆ B" and "ext_eq_on A f (g ∘ φ)"
proof -
  {
    fix a and a'
    assume "a ∈ A" and "a' ∈ A" and "φ a = φ a'"
    from this(3) have "R `` {a} = R `` {a'}" unfolding phi by simp
    with assms(1) and ‹a ∈ A› have "(a, a') ∈ R" by (rule prop_1_6_2_b)
    with ‹a ∈ A› and ‹a' ∈ A› and assms(4) have "f a = f a'" by simp
  }
  moreover from assms(1) have "A = {} ⟹ A // R = {}" by simp
  moreover note assms(3)
  ultimately obtain g where "g ` (A // R) ⊆ B" and "ext_eq_on A f (g ∘ φ)" by (elim prob_1_5_14_b)
  thus "thesis" by (fact that)
qed

proposition prob_1_6_6:
  assumes "equiv A R"
  defines phi: "φ ≡ λa. R `` {a}"
  assumes "f ` A ⊆ B"
  shows "(∃g. g ` (A // R) ⊆ B ∧ ext_eq_on A f (g ∘ φ))
            ⟷ (∀a ∈ A. ∀a' ∈ A. (a, a') ∈ R ⟶ f a = f a')"

  assume "∃g. g ` (A // R) ⊆ B ∧ ext_eq_on A f (g ∘ φ)"
