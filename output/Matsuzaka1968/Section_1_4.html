section "Various Concepts on Functions"

text ‹
  In this and the following sections, functions in usual mathematics are formalized in terms of
  functions in Isabelle/HOL. Because careful handling is required in such formalization, let me
  explain the detail.

  Functions in usual mathematics are associated with its domain and codomain. The careful handling
  is required in their formalization. A function in Isabelle/HOL (say,
  @{term [show_types] "f :: 'a ⇒ 'b"}) is associated with the domain (the universal set of the
  type @{typ "'a"}) and the codomain (the universal set of the type @{typ "'b"}). However, because
  (the universal set of) types in Isabelle/HOL are not expressive enough to encode an arbitrary
  set (which is equivalent to an arbitrary proposition under the axiom schema of comprehension in
  Isabelle/HOL), a subset of the universal set of the type @{typ "'a"} is needed to be explicitly
  specified in order to formalize the domain of a function in usual mathematics.

  As a result, a function @{term "f"} in Isabelle/HOL that formalizes a function in usual
  mathematics has "two domains"; one is the original one, i.e., the universal set of the type
  @{typ "'a"}, and the other is the one that is explicitly specified in order to express the
  domain of the function in usual mathematics.

  Let @{term "A :: 'a set"} be a set that is specified to express the domain of a function in
  usual mathematics. It should be always noted that the function in Isabelle/HOL is defined on
  @{term "-A"}. This fact is essentially different from the function in usual mathematics, where
  it is simply undefined out of @{term "A"}. This difference demands careful handling. The
  statement of definitions and propositions concerning functions in usual mathematics should be
  properly rephrased to reflect the difference.

  How each concept concerning functions in usual mathematics is rephrased in the following
  sections is named below;

    ▪The codomain is explicitly specified as a set @{term "B :: 'b set"} such that
     @{prop "f ` A ⊆ B"} only when it is actually required in definitions and propositions.
    ▪The inverse image of a set @{term "Q ⊆ B"} under @{term "f"} is defined @{term "f -` Q ∩ A"}
     since @{term "f -` Q"} could include an element out of @{term "A"}.
    ▪Surjectivity of @{term "f"} is stated as the proposition @{term "f ` A = B"}.
    ▪Injectivity of @{term "f"} is stated as the proposition @{term "inj_on f A"}.
    ▪Equality of two function @{term "f"} and @{term "g"} on a set @{term "A"} is defined by the
     proposition @{prop "∀a ∈ A. f a = g a"}, which is shortly stated as the proposition
     @{prop "ext_eq_on A f g"}. Note that @{term "f"} and @{term "g"} behaves differently out of
     @{term "A"} and thus @{prop "f = g"} does not hold even if @{prop "ext_eq_on A f g"} holds.
    ▪(TODO: inverse)

definition ext_eq_on :: "'a set ⇒ ('a ⇒ 'b) ⇒ ('a ⇒ 'b) ⇒ bool"
  where "ext_eq_on A f f' ⟷ (∀a ∈ A. f a = f' a)"

lemma ext_eq_onI [intro]:
  assumes "⋀a. a ∈ A ⟹ f a = f' a"
  shows "ext_eq_on A f f'"
  using assms unfolding ext_eq_on_def by simp

lemma ext_eq_onD [dest]:
  assumes "ext_eq_on A f f'"
    and "a ∈ A"
  shows "f a = f' a"
  using assms unfolding ext_eq_on_def by simp

lemma ext_eq_on_refl [simp]:
  shows "ext_eq_on A f f"
  by auto

lemma ext_eq_on_sym [sym]:
  assumes "ext_eq_on A f f'"
  shows "ext_eq_on A f' f"
  using assms by fastforce

lemma ext_eq_on_trans [trans]:
  assumes "ext_eq_on A f g"
    and "ext_eq_on A g h"
  shows "ext_eq_on A f h"
  using assms by fastforce

lemma ext_eq_on_empty [simp]:
  shows "ext_eq_on {} f g"
  by auto

lemma surj_onI:
  assumes "⋀a. a ∈ A ⟹ f a ∈ B"
    and "⋀b. b ∈ B ⟹ b ∈ f ` A"
  shows "f ` A = B"
proof (intro equalityI)
  from assms(1) show "f ` A ⊆ B" by auto
  from assms(2) show "B ⊆ f ` A" by auto
qed

lemma bij_betw_imageI':
  assumes "⋀a a'. a ∈ A ⟹ a' ∈ A ⟹ f a = f a' ⟹ a = a'"
    and "⋀a. a ∈ A ⟹ f a ∈ B"
    and "⋀b. b ∈ B ⟹ b ∈ f ` A"
  shows "bij_betw f A B"
proof (rule bij_betw_imageI)
  from assms(1) show "inj_on f A" by (rule inj_onI)
  from assms(2,3) show "f ` A = B" by auto
qed

lemma corr_inv_fun_fun:
  assumes "a ∈ A"
  shows "corr_inv (as_corr_on f A) (f a) = {a' ∈ A. f a' = f a}"
  using assms by auto

lemma corr_inv_fun_eq:
  shows "corr_inv (as_corr_on f A) b = {a ∈ A. f a = b}" (is "?L = ?R")
proof (rule set_eqI)
  fix a
  have "a ∈ ?L ⟷ b ∈ as_corr_on f A a" by auto
  also have "… ⟷ a ∈ A ∧ f a = b" by auto
  also have "… ⟷ a ∈ ?R" by simp
  finally show "a ∈ ?L ⟷ a ∈ ?R" .
qed

theorem thm_1_4_a:
  assumes "f ` A ⊆ B"
    and "as_corr_on g B = corr_inv (as_corr_on f A)"
  shows "bij_betw f A B"
proof (rule bij_betw_imageI')
  fix a and a'
  assume "a ∈ A"
    and "a' ∈ A"
    and "f a = f a'"
  from ‹a ∈ A› and assms(1) have "f a ∈ B" by auto
  moreover from assms(2) have "corr_inv (as_corr_on f A) (f a) = as_corr_on g B (f a)" by simp
  with ‹a ∈ A› and ‹f a ∈ B› have **: "{a'' ∈ A. f a'' = f a} = {g (f a)}"
    by (auto simp only: corr_inv_fun_fun)
  with ‹a ∈ A› have "a = g (f a)" by blast
  moreover from ** and ‹a' ∈ A› and ‹f a = f a'› have "a' = g (f a)" by auto
  ultimately show "a = a'" by simp
next
  fix a
  assume "a ∈ A"
  with assms(1) show "f a ∈ B" by auto
next
  fix b
  assume "b ∈ B"
  moreover from assms(2) have "corr_inv (as_corr_on f A) b = as_corr_on g B b" by simp
  ultimately have "{a ∈ A. f a = b} = {g b}" by (force simp only: corr_inv_fun_eq)
  hence "g b ∈ A" and "f (g b) = b" by auto
  thus "b ∈ f ` A" by force
qed

lemma bij_betwE2 [elim]:
  assumes "bij_betw f A B"
  obtains "f ` A = B"
    and "inj_on f A"
  using assms by (auto dest: bij_betw_imp_surj_on bij_betw_imp_inj_on)

theorem thm_1_4_b:
  assumes "f ` A ⊆ B"
    and "bij_betw f A B"
  obtains g where "as_corr_on g B = corr_inv (as_corr_on f A)"
proof -
  {
    fix b
    {
      assume "b ∈ B"
      from assms(2) have "f ` A = B" and "inj_on f A" by auto
      from this(1) and ‹b ∈ B› obtain a where "a ∈ A" and "b = f a" by auto
      {
        fix a'
        assume ‹a' ∈ A›
        with ‹a ∈ A› and ‹inj_on f A› have "a' = a ⟷ f a' = f a" by (simp only: inj_on_eq_iff)
        with ‹a' ∈ A› have "a' ∈ {a'. a' = a} ⟷ a' ∈ {a'. f a' = f a}" by simp
      }
      hence "{a' ∈ A. a' = a} = {a' ∈ A. f a' = f a}" by auto
      with ‹a ∈ A› have "{a'. a' = a} = {a' ∈ A. f a' = f a}" by blast
      with ‹inj_on f A› and ‹a ∈ A›
      have "{a'. a' = the_inv_into A f (f a)} = {a' ∈ A. f a' = f a}"
        by (simp only: the_inv_into_f_f)
      with ‹b = f a› have "{a. a = (the_inv_into A f) b} = {a ∈ A. f a = b}" by simp
      with ‹b ∈ B› have "as_corr_on (the_inv_into A f) B b = corr_inv (as_corr_on f A) b"
        by (force simp only: corr_inv_fun_eq)
    }
    moreover {
      assume "b ∉ B"
      with assms(1) have "{a ∈ A. f a = b} = {}" by blast
      with ‹b ∉ B› have "as_corr_on (the_inv_into A f) B b = corr_inv (as_corr_on f A) b" by blast
    }
    ultimately have "as_corr_on (the_inv_into A f) B b = corr_inv (as_corr_on f A) b" by auto
  }
  hence "as_corr_on (the_inv_into A f) B = corr_inv (as_corr_on f A)" by auto
  thus "thesis" by (fact that)
qed

theorem thm_1_4:
  assumes "f ` A ⊆ B"
  shows "(∃g. as_corr_on g B = corr_inv (as_corr_on f A)) ⟷ bij_betw f A B"
proof (rule iffI)
  assume "∃g. as_corr_on g B = corr_inv (as_corr_on f A)"
  then obtain g where "as_corr_on g B = corr_inv (as_corr_on f A)" by auto
  with assms show "bij_betw f A B" by (auto intro: thm_1_4_a)
next
  assume "bij_betw f A B"
  with assms obtain g where "as_corr_on g B = corr_inv (as_corr_on f A)" by (elim thm_1_4_b)
  thus "∃g. as_corr_on g B = corr_inv (as_corr_on f A)" by auto
qed

definition is_inv_into :: "'a set ⇒ ('a ⇒ 'b) ⇒ ('b ⇒ 'a) ⇒ bool"
  where "is_inv_into A f g ⟷ corr_inv (as_corr_on g (f ` A)) = as_corr_on f A"

lemma is_inv_intoD1:
  assumes "is_inv_into A f g"
    and "a ∈ A"
  shows "g (f a) = a"
proof -
  from assms(1) have "corr_inv (as_corr_on g (f ` A)) = as_corr_on f A" by (unfold is_inv_into_def)
  hence "corr_inv (as_corr_on g (f ` A)) a = as_corr_on f A a" by simp
  with assms(2) have "{b ∈ f ` A. g b = a} = {f a}" by (auto simp only: corr_inv_fun_eq)
  thus "g (f a) = a" by auto
qed

lemma is_inv_intoD2:
  assumes "is_inv_into A f g"
    and "b ∈ f ` A"
  shows "f (g b) = b"
  using assms by (auto dest: is_inv_intoD1)

lemma is_inv_into_imp_bij_betw1:
  assumes "is_inv_into A f g"
  shows "bij_betw f A (f ` A)"
proof -
  from assms have "corr_inv (as_corr_on g (f ` A)) = as_corr_on f A" by (unfold is_inv_into_def)
  hence "as_corr_on g (f ` A) = corr_inv (as_corr_on f A)" using prop_1_3_4 by metis
  moreover have "f ` A ⊆ f ` A" by simp
  ultimately show ?thesis by (intro thm_1_4_a)
qed

lemma is_inv_into_imp_bij_betw2:
  assumes "is_inv_into A f g"
  shows "bij_betw g (f ` A) A"
proof (rule bij_betw_imageI)
  {
    fix a
    assume "a ∈ A"
    with assms have "g (f a) = a" by (rule is_inv_intoD1)
    with ‹a ∈ A› have"g (f a) ∈ A" by simp
  }
  moreover {
    fix a
    assume "a ∈ A"
    with assms have "g (f a) = a" by (rule is_inv_intoD1)
    with ‹a ∈ A› have "∃a' ∈ A. g (f a') = a" by auto
  }
  ultimately show "g ` f ` A = A" by (blast intro: surj_onI)
  {
    fix b and b'
    assume "b ∈ f ` A"
      and "b' ∈ f ` A"
      and "g b = g b'"
    then obtain a and a' where "a ∈ A" and "b = f a" and "a' ∈ A" and "b' = f a'" by blast
    from ‹g b = g b'› and this(2,4) have "g (f a) = g (f a')" by simp
    with ‹a ∈ A› and ‹a' ∈ A› and assms have "a = a'" using is_inv_intoD1 by fastforce
    with ‹b = f a› and ‹b' = f a'› have "b = b'" by simp
  }
  thus "inj_on g (f ` A)" by (blast intro: inj_onI)
qed

theorem thm_1_5_b:

  moreover note assms(3)
  ultimately have "f a = f a'" by (elim inj_onD)
  with assms(2) and ‹a ∈ A› and ‹a' ∈ A› show "a = a'"by (elim inj_onD)
qed

proposition prob_1_4_2_b:
  assumes ― ‹The assumption @{prop "f ` A ⊆ B"} is not necessary.›

proposition prob_1_4_5_c:

  {
    assume "b ∈ f ` P"
    then obtain a' where "a' ∈ P" and "b = f a'" by auto
    from this(1) and assms(1) have "a' ∈ A" by auto
    with ‹a ∈ A› and ‹b = f a› and ‹b = f a'› and assms(2) have "a = a'" by (auto dest: inj_onD)
    with ‹a' ∈ P› and ‹a ∉ P› have "False" by simp
  }
  hence "b ∉ f ` P" by auto
  with ‹a ∈ A› and ‹b = f a› show "b ∈ f ` A - f ` P" by simp
next
  fix b
  assume "b ∈ f ` A - f ` P"
  thus "b ∈ f ` (A - P)" by blast
qed

proposition prob_1_4_8:
  assumes ― ‹The assumption @{prop "bij_betw f A B"} can be weakened to @{prop "f ` A = B"}
              because @{prop "is_inv_into A f f'"} implies @{prop "bij_betw f A (f ` A)"}.›
    "f ` A = B"
    ― ‹The assumption @{prop "bij_betw g B C"} can be weakened to @{prop "g ` B = C"} because
        @{prop "is_inv_into A f f'"} implies @{prop "bij_betw g B (g ` B)"}.›
    and "g ` B = C"
    and "is_inv_into A f f'"
    and "is_inv_into B g g'"
    and "is_inv_into A (g ∘ f) h"
  shows "ext_eq_on C h (f' ∘ g')"
proof (rule ext_eq_onI)
  fix c
  assume "c ∈ C"
  from assms(5) have "bij_betw h ((g ∘ f) ` A) A" by (intro is_inv_into_imp_bij_betw2)
  moreover from assms(1,2) have "(g ∘ f) ` A = C" by auto
  ultimately have "bij_betw h C A" by simp
  with ‹c ∈ C› have "h c ∈ A" by (auto dest: bij_betw_imp_surj_on)
  from ‹c ∈ C› and ‹(g ∘ f) ` A = C› have "c ∈ (g ∘ f) ` A" by simp
  with assms(5) have "g (f (h c)) = c" by (auto dest: is_inv_intoD2)
  hence "g' (g (f (h c))) = g' c" by simp
  moreover from ‹h c ∈ A› and assms(1) have "f (h c) ∈ B" by auto
  moreover note assms(4)
  ultimately have "f (h c) = g' c" by (auto dest: is_inv_intoD1)
  hence "f' (f (h c)) = f' (g' c)" by simp
  moreover note ‹h c ∈ A›
  moreover note assms(3)
  ultimately have "h c = f' (g' c)" by (auto dest: is_inv_intoD1)
  thus "h c = (f' ∘ g') c" by simp
qed

proposition prob_1_4_13_a:
  assumes "f ` A ⊆ B"
    and "g ` B ⊆ C"
    and "(g ∘ f) ` A = C"
    and "inj_on g B"
  shows "f ` A = B"
proof (rule surj_onI)
  fix a
  assume "a ∈ A"
  with assms(1) show "f a ∈ B" by auto
next
  fix b
  assume "b ∈ B"
  with assms(2) have "g b ∈ C" by auto
  with assms(3) obtain a where "a ∈ A" and "g (f a) = g b" by force
  moreover from this(1) and assms(1) have "f a ∈ B" by auto
  moreover note ‹b ∈ B› and assms(4)
  ultimately have "f a = b" by (elim inj_onD)
  with ‹a ∈ A› show "b ∈ f ` A" by auto
qed

proposition prob_1_4_13_b:
  assumes ― ‹The assumption @{prop "g ` B ⊆ C"} is not necessary.›
    "inj_on (g ∘ f) A"
    and "f ` A = B"
  shows "inj_on g B"
proof (rule inj_onI)
  fix b b'
  assume "b ∈ B" and "b' ∈ B" and "g b = g b'"
  from this(1,2) and assms(2) obtain a and a'
    where "a ∈ A" and "b = f a" and "a' ∈ A" and "b' = f a'" by auto
  from ‹g b = g b'› and this(2,4) have "g (f a) = g (f a')" by simp
  with ‹a ∈ A› and ‹a' ∈ A› and assms(1) have "a = a'" by (auto elim: inj_onD)
  with ‹b = f a› and ‹b' = f a'› show "b = b'" by simp
qed

proposition prob_1_4_14:
  assumes "f ` A ⊆ B"
    and "g ` B ⊆ A"
    and "g' ` B ⊆ A"
    and "id_on (g ∘ f) A"
    and "id_on (f ∘ g') B"
    and "is_inv_into A f f'"
  obtains "bij_betw f A B"
    and "ext_eq_on B g g'"
    and "ext_eq_on B g' f'"
proof -
  have "bij_betw f A B"
  proof (rule bij_betw_imageI)
    {
      fix a and a'
      assume "a ∈ A" and "a' ∈ A" and "f a = f a'"
      from this(3) have "g (f a) = g (f a')" by simp
      with ‹a ∈ A› and ‹a' ∈ A› and assms(4) have "a = a'" using id_onD by fastforce
    }
    thus "inj_on f A" by (intro inj_onI)
    {
      fix b
      assume "b ∈ B"
      with assms(5) have "f (g' b) = b" by auto
      moreover from assms(3) and ‹b ∈ B› have "g' b ∈ A" by auto
      ultimately have "b ∈ f ` A" by force
    }
    hence "B ⊆ f ` A" ..
    with assms(1) show "f ` A = B" by simp
  qed
  moreover have "ext_eq_on B g g'"
  proof (rule ext_eq_onI)
    fix b
    assume "b ∈ B"
    with assms(5) have "f (g' b) = b" by auto
    hence "g (f (g' b)) = g b" by simp
    moreover from ‹b ∈ B› and assms(3) have "g' b ∈ A" by auto
    moreover note assms(4)
    ultimately show "g b = g' b" by auto
  qed
  moreover have "ext_eq_on B g' f'"
  proof (rule ext_eq_onI)
    fix b
    assume "b ∈ B"
    with assms(5) have "f (g' b) = b" by auto
    hence "f' (f (g' b)) = f' b" by simp
    moreover from ‹b ∈ B› and assms(3) have "g' b ∈ A" by auto
    moreover note assms(6)
    ultimately show "g' b = f' b" by (auto dest: is_inv_intoD1)
  qed
  ultimately show "thesis" by (fact that)
qed

abbreviation χ :: "'a set ⇒ 'a ⇒ int"
  where "χ ≡ indicator"

lemma indicator_definition:
  obtains "x ∈ X ⟹ χ X x = 1"
  | "x ∉ X ⟹ χ X x = 0"
  by simp

lemma chi_0_1:
  shows "χ A a ∈ {0, 1}"
proof -
  {
    assume "a ∈ A"
    hence "χ A a = 1" by simp
  }
  moreover {
    assume "a ∉ A"
    hence "χ A a = 0" by simp
  }
  ultimately show "χ A a ∈ {0, 1}" by blast
qed

lemma one_leq_chiD:
  assumes "1 ≤ χ A a"
  shows "a ∈ A"
proof -
  from assms and chi_0_1 have "χ A a = 1" by force
  thus ?thesis by (simp only: indicator_eq_1_iff)
qed

proposition prob_1_4_15_0_a:
  assumes "A ⊆ X"
    and "B ⊆ X"
    and "⋀x. x ∈ X ⟹ χ A x ≤ χ B x"
  shows "A ⊆ B"
proof (rule subsetI)
  fix a
  assume "a ∈ A"
  hence "1 = χ A a" by simp
  also have "… ≤ χ B a"
  proof -
    from ‹a ∈ A› and assms(1) have "a ∈ X" by auto
    with assms(3) show ?thesis by simp
  qed
  finally have "1 ≤ χ B a" by simp
  thus "a ∈ B" by (auto dest: one_leq_chiD)
qed

proposition prob_1_4_15_0_b:
  assumes "A ⊆ X"
    and "B ⊆ X"
    and "A ⊆ B"
    and "x ∈ X"
  shows "χ A x ≤ χ B x"
proof -
  from assms consider (A) "x ∈ A" | (B) "x ∈ B - A" | (X) "x ∈ X - B" by auto
  thus ?thesis
  proof cases
    case A
    with assms(3) have "x ∈ A" and "x ∈ B" by auto
    thus ?thesis by simp
  next
    case B
    hence "x ∉ A" and "x ∈ B" by auto
    thus ?thesis by simp
  next
    case X
    with assms(3) have "x ∉ A" and "x ∉ B" by auto
    thus ?thesis by simp
  qed
qed

proposition prob_1_4_15:
  assumes "A ⊆ X"
    and "B ⊆ X"
  shows "(∀x ∈ X. χ A x ≤ χ B x) ⟷ A ⊆ B"

  with assms show "∀x ∈ X. χ A x ≤ χ B x" using prob_1_4_15_0_b by metis
qed
