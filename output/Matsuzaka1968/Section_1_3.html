subsection ‹B) Notion of Correspondence›

text ‹
  In this section, a correspondence is defined by a function of the type @{typ "'a => 'b set"}.
  Neither the initial nor target set of a correspondence is not explicitly specified. Instead, for
  a correspondence @{term "Γ"}, its initial set is implicitly specified by a set
  @{term [show_types] "A :: 'a set"} such that @{prop "∀a. a ∉ A ⟶ Γ a = {}"}, and its target
  set is implicitly specified by a set @{term "B :: 'b set"} such that @{prop "Γ `` A ⊆ B"}.
›

subsection ‹C) Graph of Correspondence›

definition corr_graph :: "('a ⇒ 'b set) ⇒ ('a × 'b) set"
  where "corr_graph Γ = {(a, b). b ∈ Γ a}"

lemma corr_graphI [intro]:
  assumes "b ∈ Γ a"
  shows "(a, b) ∈ corr_graph Γ"
  using assms unfolding corr_graph_def by simp

lemma corr_graphD [dest]:
  assumes "(a, b) ∈ corr_graph Γ"
  shows "b ∈ Γ a"
  using assms unfolding corr_graph_def by simp

proposition prop_1_3_1:
  shows "Γ a = {b. (a, b) ∈ corr_graph Γ}"
  by auto

theorem thm_1_1:
  shows "∃!Γ. corr_graph Γ = G"
proof -
  define Γ where "Γ a ≡ {b. (a, b) ∈ G}" for a
  show ?thesis
  proof (rule ex1I)
    from Γ_def show "corr_graph Γ = G" by auto
  next
    fix Γ'
    assume "corr_graph Γ' = G"
    with Γ_def show "Γ' = Γ" by auto
  qed
qed

lemma corr_graph_inject:
  assumes "corr_graph Γ = corr_graph Γ'"
  shows "Γ = Γ'"
  using assms thm_1_1 by auto

definition corr_dom :: "('a ⇒ 'b set) ⇒ 'a set"
  where "corr_dom Γ = {a. ∃b. (a, b) ∈ corr_graph Γ}"

lemma corr_domI [intro]:

lemma corr_domE [elim]:

definition corr_range :: "('a ⇒ 'b set) ⇒ 'b set"

lemma corr_rangeI [intro]:

subsection ‹E) Maps›

definition as_corr_on :: "('a ⇒ 'b) ⇒ 'a set ⇒ 'a ⇒ 'b set"
  where "as_corr_on f A a = (if a ∈ A then {f a} else {})"

text ‹
  @{const "as_corr_on"} transforms a function into the correspondence. Since a function
  @{term "f :: 'a ⇒ 'b"} in Isabelle/HOL is total, which means that the function is always
  defined on the universal set @{term "UNIV :: 'a set"} of the type @{typ "'a"},
  @{const "as_corr_on"} additionally specifies a set of the type @{typ "'a set"} that should be
  considered as the domain of the function.
›

lemma as_corr_onI [intro]:
  assumes "a ∈ A"
    and "f a = b"
  shows "b ∈ as_corr_on f A a"
  using assms unfolding as_corr_on_def by simp

lemma as_corr_onE [elim]:
  assumes "b ∈ as_corr_on f A a"
  obtains "a ∈ A" and "f a = b"
proof -
  from assms have *: "b ∈ (if a ∈ A then {f a} else {})" by (simp only: as_corr_on_def)
  {
    assume "a ∉ A"
    with * have "False" by simp
  }
  hence "a ∈ A" by auto
  moreover from this and * have "f a = b" by simp
  ultimately show "thesis" by (fact that)
qed

definition corr_functional_on :: "('a ⇒ 'b set) ⇒ 'a set ⇒ bool"
  where "corr_functional_on Γ A ⟷ (∀a. (a ∈ A ⟶ (∃!b. b ∈ Γ a)) ∧ (a ∉ A ⟶ Γ a = {}))"

definition id_on :: "('a ⇒ 'a) ⇒ 'a set ⇒ bool"
  where "id_on f A ⟷ (∀a ∈ A. f a = a)"

text ‹
  @{prop "id_on f A"} states that the function @{term f} behaves as an identity function on
  the set @{term A}. This proposition does not specify any other property of @{term f} on an
  element out of @{term A}.
›

lemmas id_on_iff = id_on_def

lemma id_onI [intro]:
  assumes "⋀a. a ∈ A ⟹ f a = a"
  shows "id_on f A"
  using assms unfolding id_on_def by simp

lemma id_onD [dest]:
  assumes "id_on f A"
    and "a ∈ A"
  shows "f a = a"
  using assms unfolding id_on_def by fast

lemma id_on_empty [simp]:
  shows "id_on f {}"
  by auto

lemma set_eqI2:
  assumes "⋀x. x ∈ A ⟹ x ∈ B"
    and "⋀x. x ∈ B ⟹ x ∈ A"
  shows "A = B"
  using assms by blast

lemma id_on_imp_surj_on:
  assumes "id_on f A"
  shows "f ` A = A"
  using assms by force

lemma id_on_imp_inj_on:
  assumes "id_on f A"
  shows "inj_on f A"
proof (rule inj_onI)
  fix a and a'
  assume "a ∈ A" and "a' ∈ A" and "f a = f a'"
  with assms show "a = a'" using id_onD by fastforce
qed

lemma id_on_imp_bij_betw:
  assumes "id_on f A"
  shows "bij_betw f A A"
  using assms by (auto intro: bij_betw_imageI dest: id_on_imp_surj_on id_on_imp_inj_on)

lemma thm_1_2_a:
  assumes ― ‹ The assumption @{prop "G ⊆ A × B"} is not necessary.›
    "f ` A ⊆ B"
    and "corr_graph (as_corr_on f A) = G"
  shows "∀a ∈ A. ∃!b ∈ B. (a, b) ∈ G"
  using assms by fast

lemma thm_1_2_b:
  assumes "G ⊆ A × B"
    and "∀a ∈ A. ∃!b ∈ B. (a, b) ∈ G"
  obtains f where
    "f ` A ⊆ B"
    and "corr_graph (as_corr_on f A) = G"
proof -
  define f where "f a ≡ THE b. b ∈ B ∧ (a, b) ∈ G" for a
  have f0: "f a ∈ B" and f1: "(a, f a) ∈ G" if "a ∈ A" for a
  proof -
    from assms(2) and that have "∃!b. b ∈ B ∧ (a, b) ∈ G" by simp
    hence "f a ∈ B ∧ (a, f a) ∈ G" (is "?L ∧ ?R") unfolding f_def by (fact theI')
    thus ?L and ?R by simp+
  qed
  from f0 have "f ` A ⊆ B" by auto
  moreover have "corr_graph (as_corr_on f A) = G"
  proof (rule set_eqI2, split_pair; split_pair)
    fix a b
    assume "(a, b) ∈ corr_graph (as_corr_on f A)"
    hence "b ∈ as_corr_on f A a" by auto
    hence "a ∈ A" and "f a = b" by auto
    thus "(a, b) ∈ G" by (auto intro: f1)
  next
    fix a b
    assume "(a, b) ∈ G"
    with assms(1) have "a ∈ A" and "b ∈ B" by auto
    from this(1) and assms(2) have "∃!b ∈ B. (a, b) ∈ G" by simp
    with ‹b ∈ B› and ‹(a, b) ∈ G› have "f a = b" unfolding f_def by auto
    with ‹a ∈ A› have "b ∈ as_corr_on f A a" by auto
    thus "(a, b) ∈ corr_graph (as_corr_on f A)" by auto
  qed
  ultimately show thesis by (fact that)
qed

theorem thm_1_2:
  assumes "G ⊆ A × B"
  shows "(∃f. f ` A ⊆ B ∧ corr_graph (as_corr_on f A) = G) ⟷ (∀a ∈ A. ∃!b ∈ B. (a, b) ∈ G)"
    (is "?L ⟷ ?R")
proof (intro iffI)
  assume "?L"
  then obtain f where "f ` A ⊆ B" and "corr_graph (as_corr_on f A) = G" by blast
  thus "?R" by (elim thm_1_2_a)
next
  assume "?R"
  with assms obtain f where "f ` A ⊆ B" and "corr_graph (as_corr_on f A) = G" by (elim thm_1_2_b)
  thus "?L" by auto
qed

proposition prob_1_3_3_a:
  assumes ― ‹The assumption @{prop "corr_graph Γ ⊆ A × B"} is not necessary.›
    ― ‹Since @{prop "corr_dom Γ = A"}* is an assumption, the assumption
        @{prop "corr_graph Γ ⊆ A × B"} can be replaced by @{prop "corr_range Γ ⊆ B"}.›
    "corr_range Γ ⊆ B"
    and "corr_dom Γ = A"
    and "⋀b b'. b ∈ B ⟹ b' ∈ B ⟹ b ≠ b' ⟹ corr_inv Γ b ∩ corr_inv Γ b' = {}"
  obtains f where "f ` A ⊆ B" and "as_corr_on f A = Γ"
proof -
  {
    fix a
    assume "a ∈ A"
    with assms(2) obtain b where *: "b ∈ Γ a" by auto
    with assms(1) have "b ∈ B" by auto
    moreover from * have "(a, b) ∈ corr_graph Γ" by auto
    moreover {
      fix b'
      assume "b' ∈ B" and "(a, b') ∈ corr_graph Γ"
      {
        assume "b' ≠ b"
        moreover note ‹b ∈ B› and ‹b' ∈ B›
        moreover have "corr_inv Γ b ∩ corr_inv Γ b' ≠ {}"
        proof -
          from ‹b ∈ Γ a› have "a ∈ corr_inv Γ b" by auto
          moreover from ‹(a, b') ∈ corr_graph Γ› have "a ∈ corr_inv Γ b'" by auto
          ultimately show ?thesis by auto
        qed
        ultimately have "False" using assms(3) by simp
      }
      hence "b' = b" by auto
    }
    ultimately have "∃!b ∈ B. (a, b) ∈ corr_graph Γ" by blast
  }
  hence "∀a ∈ A. ∃!b ∈ B. (a, b) ∈ corr_graph Γ" ..
  moreover from assms(1,2) have "corr_graph Γ ⊆ A × B" by auto
  ultimately obtain f
    where "f ` A ⊆ B"
      and "corr_graph (as_corr_on f A) = corr_graph Γ" by (elim thm_1_2_b)
  hence "as_corr_on f A = Γ" by (elim corr_graph_inject)
  with ‹f ` A ⊆ B› show "thesis" by (rule that)
qed

proposition prob_1_3_3_b:
  assumes ― ‹The assumption @{prop "corr_graph Γ ⊆ A × B"} is not necessary.›
    ― ‹Original assumptions would include @{prop "corr_graph Γ ⊆ A × B"} but it can be weakened
        to the assumption @{prop "corr_dom Γ ⊆ A"}.›
    "corr_dom Γ ⊆ A"
    and "f ` A ⊆ B"
    and "as_corr_on f A = Γ"
  obtains "corr_dom Γ = A"
    and "⋀b b'. b ∈ B ⟹ b' ∈ B ⟹ b ≠ b' ⟹ corr_inv Γ b ∩ corr_inv Γ b' = {}"
proof -
  from assms(3) have "corr_graph (as_corr_on f A) = corr_graph Γ" by simp
  with assms(2) have *: "∀a ∈ A. ∃!b ∈ B. (a, b) ∈ corr_graph Γ" by (rule thm_1_2_a)
  note assms(1)
  moreover have "A ⊆ corr_dom Γ" using * by blast
  ultimately have "corr_dom Γ = A" ..
  moreover {
    fix b and b'
    assume "b ∈ B" and "b' ∈ B" and "b ≠ b'"
    {
      fix a
      assume "a ∈ corr_inv Γ b" and "a ∈ corr_inv Γ b'"
      hence "(a, b) ∈ corr_graph Γ" and "(a, b') ∈ corr_graph Γ" by auto
      moreover from ‹(a, b) ∈ corr_graph Γ› and assms(1) have "a ∈ A" by auto
      moreover note ‹b ∈ B› and ‹b' ∈ B›
      ultimately have "b = b'" using * by auto
      with ‹b ≠ b'› have "False" ..
    }
    hence "corr_inv Γ b ∩ corr_inv Γ b' = {}" by auto
  }
  ultimately show "thesis" by (fact that)
qed
